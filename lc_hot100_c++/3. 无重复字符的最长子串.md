class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        int n = s.size(), res = 0;
        unordered_map<char, int> mp;

        for (int i = 0, j = 0; i < n; i++)
        {
            mp[s[i]]++;
            while (j <= i && mp[s[i]] > 1)
            {
                mp[s[j]]--;
                j++;
            }

            res = max(i - j + 1, res);
        }

        return res;
    }
};
