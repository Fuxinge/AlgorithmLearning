```python
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        if not strs:
            return ""

        # 外层循环：遍历第一个字符串的每一个字符索引 (假定它就是基准)
        for i in range(len(strs[0])):
            tar = strs[0][i]

            # 内层循环：拿这个字符，去和其他所有字符串的同一位置比较
            for j in range(1, len(strs)):
                # 触发结束的两个条件：
                # 1. 某个字符串太短，已经越界了 (i == len(strs[j]))
                # 2. 发现当前位置的字符不一样了
                if i == len(strs[j]) or strs[j][i] != tar:
                    # 直接截取第一个字符串从 0 到 i 的部分，这就是最长前缀！
                    return strs[0][:i]

        # 如果外层循环完美结束，说明第一个字符串本身就是公共前缀
        return strs[0]
```
