```python
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        n = len(prices)
        dp = [[0] * 2 for _ in range(n + 1)]

        dp[0][1] = -inf
        for i in range(n):
            dp[i+1][0] = max(dp[i][0],dp[i][1] + prices[i])
            dp[i+1][1] = max(dp[i][1],dp[i][0] - prices[i])

        return dp[n][0]
```
