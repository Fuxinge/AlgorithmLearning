```python
class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        n = k = len(nums)

        i = 0
        j = n - 1
        while i <= j:
            if nums[i] == val:
                nums[i],nums[j] = nums[j],nums[i]
                j -= 1
                k -= 1
            else :
                i += 1

        return k
```
