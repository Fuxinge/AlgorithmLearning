```python
class Solution:
    def candy(self, nums: List[int]) -> int:
        n = len(nums)
        res = n
        i = 0
        while i < n :
            start = i - 1 if i > 0 and nums[i-1] < nums[i] else i

            while i + 1 < n and nums[i] < nums[i + 1]:
                i += 1
            top = i

            while i + 1 < n and nums[i] > nums[i + 1]:
                i += 1

            inc = top - start
            dec = i - top
            res += (inc * (inc - 1) + dec * (dec - 1)) // 2 + max(inc, dec)
            i += 1

        return res
```
