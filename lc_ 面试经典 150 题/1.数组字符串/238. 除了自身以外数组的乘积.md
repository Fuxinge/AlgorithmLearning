```python
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        nxt = [1] * n
        for i in range(n-2,-1,-1):
            nxt[i] = nxt[i+1] * nums[i + 1]

        pre = [1] * n
        for i in range(1,n):
            pre[i] = pre[i-1] * nums[i-1]

        for i in range(n) :
            nums[i] = pre[i] * nxt[i]

        return nums
```
