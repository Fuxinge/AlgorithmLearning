class RandomizedSet:

    def __init__(self):
        self.mp = defaultdict(int)
        self.lt = []

    def insert(self, val: int) -> bool:
        if val in self.mp:
            return False
        else :
            self.lt.append(val)
            self.mp[val] = len(self.lt) - 1
            return True

    def remove(self, val: int) -> bool:
        if val in self.mp:
            index = self.mp[val]
            last_val = self.lt[-1]
            self.lt[index] = last_val
            self.mp[last_val] = index
            self.lt.pop()
            del self.mp[val]
            return True
        else :
            return False

    def getRandom(self) -> int:
        return choice(self.lt)
