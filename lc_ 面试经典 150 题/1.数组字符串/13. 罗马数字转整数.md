class Solution:
    def romanToInt(self, s: str) -> int:
        mp = {}
        mp['I'] = 1
        mp['V'] = 5
        mp['X'] = 10
        mp['L'] = 50
        mp['C'] = 100
        mp['D'] = 500
        mp['M'] = 1000

        res = 0
        for x,y in pairwise(s):
            x,y = mp[x],mp[y]
            res += x if x >= y else -x
        return res + mp[s[-1]]
