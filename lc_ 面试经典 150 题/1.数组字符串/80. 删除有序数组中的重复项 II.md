```python
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        n = len(nums)
        k = 0
        tmp = 1
        i = 0
        while i < n:
            while i + 1 < n and nums[i] == nums[i + 1]:
                tmp += 1
                i += 1
            if tmp <= 2:
                while tmp:
                    nums[k] = nums[i]
                    k += 1
                    tmp -= 1
            else :
                nums[k] = nums[i]
                nums[k+1] = nums[i]
                k += 2
            i += 1
            tmp = 1
        return k
```
