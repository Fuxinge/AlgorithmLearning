```python
class Solution:
    def jump(self, nums: List[int]) -> int:
        res = 0
        max_right = 0
        cur = 0
        n = len(nums)
        for i in range(n-1):
            c = nums[i]
            max_right = max(max_right,i + c)

            if i == cur:
                res += 1
                cur = max_right

        return res
```
