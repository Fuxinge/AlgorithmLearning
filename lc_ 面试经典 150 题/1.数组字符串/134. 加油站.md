class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        s = min_s = res = 0
        n = len(cost)
        for i,(g,c) in enumerate(zip(gas,cost)):
            s += g - c
            if s < min_s:
                min_s = s
                res = i + 1
        return -1 if s < 0 else res
