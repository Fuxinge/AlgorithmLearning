```python
class Solution:
    def trap(self, height: List[int]) -> int:
        n = len(height)
        res = 0
        left = [height[0]] * n
        right = [height[-1]] * n
        for i,c in enumerate(height):
            left[i] = max(left[i-1],c)
        for i in range(n-2,-1,-1):
            right[i] = max(right[i+1],height[i])

        for i in range(n):
            res += min(left[i],right[i]) - height[i]

        return res
```
