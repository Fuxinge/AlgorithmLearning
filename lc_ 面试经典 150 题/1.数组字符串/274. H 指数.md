class Solution:
    def hIndex(self, citations: List[int]) -> int:
        n = len(citations)
        s = 0
        cnt = [0] * ( n + 1)
        for c in citations:
            cnt[min(c,n)] += 1
        for i in range(n,-1,-1):
            s+=cnt[i]
            if s >= i:
                return i
