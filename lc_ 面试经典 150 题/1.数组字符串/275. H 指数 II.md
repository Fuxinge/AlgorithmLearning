```python
class Solution:
    def hIndex(self, citations: List[int]) -> int:
        def check(mid,citations):
            return citations[-mid] >= mid

        n = len(citations)
        l,r = -1,n + 1
        while l + 1 < r:
            mid = (l + r) // 2
            if check(mid,citations):
                l = mid
            else :
                r = mid

        return l
```
