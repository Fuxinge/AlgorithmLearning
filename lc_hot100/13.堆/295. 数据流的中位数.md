class MedianFinder {
    priority_queue<int> left;
    priority_queue<int,vector<int> ,greater<int> > right;
public:
    MedianFinder() {

    }

    void addNum(int num) {
        int n = left.size(),m = right.size();
        if((n+m) %2)
        {
            left.push(num);
            auto t = left.top();
            left.pop();
            right.push(t);
        }
        else
        {
            right.push(num);
            auto t = right.top();
            right.pop();
            left.push(t);
        }
    }

    double findMedian() {
        int n = left.size(),m = right.size();
        if(( n + m ) % 2)
        {
            return left.top() * 1.0;
        }
        else
        {
            return (left.top() + right.top()) * 0.5;
        }
    }
};
