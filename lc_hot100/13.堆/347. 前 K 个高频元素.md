```cpp
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        vector<pair<int,int>> temp;
        vector<int> res;
        int n = nums.size();
        unordered_map<int,int> mp;

        for(auto x:nums)
        {
            mp[x]++;
        }

        for(auto x: mp)
        {
            temp.push_back({x.second,x.first});
        }

        sort(temp.begin(),temp.end());
        reverse(temp.begin(),temp.end());
        for(int i = 0;i<k;i++)
        {
            res.push_back(temp[i].second);
        }

        return res;
    }
};
```

基于桶排序O(n)时间复杂度

```cpp
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        vector<pair<int,int>> temp;
        vector<int> res;
        int n = nums.size();
        unordered_map<int,int> mp;

        int max_cnt = 0;

        for(auto x:nums)
        {
            mp[x]++;
            max_cnt = max(max_cnt,mp[x]);
        }

        vector<vector<int>> buckets(max_cnt + 1);

        for(auto x:mp)
        {
            buckets[x.second].push_back(x.first);
        }

        for(int i = max_cnt;i>=0 && res.size() < k ;i--)
        {
            res.insert(res.end(),buckets[i].begin(),buckets[i].end());
        }

        return res;
    }
};
```

```python
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        cnt = Counter(nums)
        max_cnt = max(cnt.values())

        bucket = [[] for _ in range(max_cnt + 1)]

        for a,b in cnt.items():
            bucket[b].append(a)

        res = []

        for b in reversed(bucket):
            res += b
            if len(res) == k:
                return res
```
