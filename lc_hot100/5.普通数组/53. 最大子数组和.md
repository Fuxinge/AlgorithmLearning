class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        int res = nums[0];

        // 以i结尾的最大子数组和
        vector<int> dp(n+1,0);

        dp[0] = nums[0];
        for(int i = 1;i<n;i++)
        {
            dp[i] = max(nums[i],dp[i-1] + nums[i]);
            res = max(res,dp[i]);
        }

        return res;
    }
};

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        dp = [0] * (len(nums))
        dp[0] = nums[0]
        for i in range(1,len(nums)):
            dp[i] = max(dp[i-1] + nums[i],nums[i])

        res = max(dp)

        return res
