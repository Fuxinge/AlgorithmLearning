class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        vector<vector<int>> res;
        vector<pair<int,int>> v;
        for(auto a:intervals)
        {
            v.push_back({a[0],a[1]});
        }

        sort(v.begin(),v.end());
        int n = v.size();
        int st = INT_MIN,ed = INT_MIN;
        for(int i = 0;i < n;i ++ )
        {
            int s = v[i].first,e = v[i].second;
            if(st==INT_MIN)
            {
                st = s,ed = e;
            }
            else if(s>ed)
            {
                res.push_back({st,ed});
                st = s,ed = e;
            }
            else
            {
                ed = max(ed,e);
            }
        }
        if(st!=INT_MIN) res.push_back({st,ed});

        return res;
    }
};

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort(key = lambda p:p[0])
        n = len(intervals)
        res = []
        st,ed = -1,-1
        for i in range(n):
            ST,ED = intervals[i][0],intervals[i][1]
            if st == -1:
                st = ST
                ed = ED
            elif ST > ed:
                res.append([st,ed])
                st = ST
                ed = ED
            else :
                ed = max(ED,ed)
        if st != -1:
            res.append([st,ed])

        return res
