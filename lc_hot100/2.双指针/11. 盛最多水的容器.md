class Solution {
public:
    int maxArea(vector<int>& height) {
        int n = height.size();
        int res = INT_MIN;
        int i = 0,j = n - 1;
        while(i<j)
        {
            res = max(res,min(height[i],height[j])*(j-i));
            if(height[i]<height[j])
            {
                i++;
            }
            else j--;
        }

        return res;
    }
};

class Solution:
    def maxArea(self, height: List[int]) -> int:
        res = -1
        j = len(height) - 1
        i = 0
        while i<j:
            res = max((j - i) * min(height[i],height[j]),res)
            if height[i] < height[j]:
                i = i + 1
            else :
                j = j - 1


        return res
