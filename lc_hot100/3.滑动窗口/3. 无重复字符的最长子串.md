```cpp
class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        int n = s.size(), res = 0;
        unordered_map<char, int> mp;

        for (int i = 0, j = 0; i < n; i++)
        {
            mp[s[i]]++;
            while (j <= i && mp[s[i]] > 1)
            {
                mp[s[j]]--;
                j++;
            }

            res = max(i - j + 1, res);
        }

        return res;
    }
};
```

```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        mp = defaultdict(int)
        j = 0
        res = 0
        for i,c in enumerate(s):
            mp[c] += 1
            while j <= i and mp[s[i]] > 1:
                mp[s[j]] -= 1
                j = j + 1
            res = max(res, i - j + 1)
        return res
```
