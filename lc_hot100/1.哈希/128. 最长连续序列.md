```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution
{
public:
    int longestConsecutive(vector<int> &nums)
    {
        int n = nums.size();
        int res = 0;
        unordered_set<int> st;
        for (auto x : nums)
        {
            st.insert(x);
        }
        for (auto x : st)
        {
            if (st.find(x - 1) != st.end())
                continue;

            int y = x + 1;
            while (st.find(y) != st.end())
            {
                y++;
            }

            res = max(res, y - x);
        }

        return res;
    }
};
```

```python
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        res = 0
        st = set(nums)
        for i in st:
            if i - 1 in st:
                continue
            y = i + 1
            while y in st:
                y = y + 1

            res = max(res,y - i)

        return res
```
