需要额外注意return里的 l<m\*n

```cpp
class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int n = matrix.size(),m = matrix[0].size();
        int l = 0,r = n * m - 1;
        while(l<=r)
        {
            int mid = (l + r) / 2;
            int row = mid / m , col = mid % m;
            if(matrix[row][col] < target)
            {
                l = mid + 1;
            }
            else r = mid - 1;
        }

        return l<m*n && matrix[l / m][l % m] == target;
    }
};
```
