class Solution {
public:
    int findMin(vector<int>& nums) {
        int n = nums.size();
        if(n==1) return 0;
        int l = 0, r = n - 1;
        while(l<=r)
        {
            int mid = ( l + r) / 2;
            if(nums[mid] <= nums.back())
            {
                r = mid - 1;
            }
            else l = mid + 1;
        }

        return l;
    }
    int search(vector<int>& nums, int target) {
        int n = nums.size();
        int k = findMin(nums);

        if(target < nums.back())
        {
            auto it = lower_bound(nums.begin() + k ,nums.end(),target);
            int st = it - nums.begin();
            if(st == n || nums[st] != target) return -1;
            return st;
        }
        else if(target == nums.back()) return n - 1;
        else
        {
            auto it = lower_bound(nums.begin(),nums.begin() + k,target);
            int st = it - nums.begin();
            if(st == k || nums[st] != target) return -1;
            return st;
        }
    }
};
