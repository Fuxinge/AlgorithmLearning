闭区间写法 只能返回left

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int l = 0,r = n - 1;
        while(l<=r)
        {
            int mid = (l + r) / 2;
            if(nums[mid] < target)
            {
                l = mid + 1;
            }
            else r = mid - 1;
        }

        return l;
    }
};
```

左闭右开区间写法 返回left，right均可

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int l = 0, r = n;
        while(l<r)
        {
            int mid = (l + r) / 2;
            if(nums[mid] < target) l = mid + 1;
            else r = mid;
        }

        return l;
    }
};
```

开区间写法 只能返回right

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int l = -1, r = n;
        while(l + 1 < r)
        {
            int mid =( l + r ) / 2;
            if(nums[mid] < target) l = mid;
            else r = mid;
        }

        return r;
    }
};
```
