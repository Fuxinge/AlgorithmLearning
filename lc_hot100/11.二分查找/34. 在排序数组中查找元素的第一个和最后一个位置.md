class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();

        auto it = lower_bound(nums.begin(),nums.end(),target);
        int st = it - nums.begin();

        if(st==n || nums[st] != target)
        {
            return {-1,-1};
        }

        auto itt = upper_bound(nums.begin(),nums.end(),target);
        int ed = itt - nums.begin() - 1;
        return  {st,ed};


    }
};

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        st = bisect_left(nums,target)

        if st == len(nums) or nums[st] != target:
            return [-1,-1]
        ed = bisect_right(nums,target) - 1

        return [st,ed]

闭区间正常写法

```cpp
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();
        int l = 0,r = n -1;
        while(l<=r)
        {
            int mid = (l + r) / 2;
            if(nums[mid] < target) l = mid + 1;
            else r = mid - 1;
        }

        if(l==n || nums[l] != target)
            return {-1,-1};
        int st = l;
        l = 0,r = n-1;
        while(l<=r)
        {
            int mid = (l+r)/2;
            if(nums[mid] < target + 1) l = mid + 1;
            else r = mid - 1;
        }

        return {st,l-1};
    }
};
```

```python
class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        n = len(nums)
        l,r = 0,n - 1

        st = -1
        ed = -1

        while l <= r:
            mid = (l + r) // 2
            if nums[mid] < target:
                l = mid + 1
            else:
                r = mid - 1

        st = l
        if st == n or nums[st] != target:
            return [-1,-1]

        l = 0
        r = n - 1
        while l <= r:
            mid = (l + r) >> 1
            if nums[mid] < target + 1:
                l = mid + 1
            else:
                r = mid - 1

        ed = l  -1

        return [st,ed]
```
