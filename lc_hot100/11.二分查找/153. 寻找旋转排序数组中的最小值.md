class Solution {
public:
    int findMin(vector<int>& nums) {
        int n = nums.size();
        if(n==1) return nums[0];
        int l = 0, r = n - 1;
        while(l<=r)
        {
            int mid = ( l + r) / 2;
            if(nums[mid] <= nums.back())
            {
                r = mid - 1;
            }
            else l = mid + 1;
        }

        return nums[l];
    }


};

class Solution:
    def findMin(self, nums: List[int]) -> int:
        n = len(nums)
        l,r = 0 ,n - 1

        while l<=r:
            mid = (l+r) //2
            if nums[mid] <= nums[-1]:
                r = mid - 1
            else :
                l = mid + 1

        low = l

        return nums[low]
