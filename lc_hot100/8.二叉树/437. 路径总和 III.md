# 方法一：双重递归（深度优先搜索）

这种方法的核心思想是遍历二叉树的每个节点，并以每个节点作为起点，分别计算向下延伸的路径中满足条件的数目。

算法流程：

1. 外层递归 (pathSum)：遍历每个节点，计算以当前节点为起点的路径数，并递归处理左右子树。
2. 内层递归 (rootSum)：从给定节点出发，向下搜索路径。如果当前路径和等于目标值，计数加1，并继续在左右子树中搜索（目标值减去当前节点值）。

```cpp
class Solution {
public:
    int pathSum(TreeNode* root, int targetSum) {
        if (!root) return 0;
        // 当前节点为起点的路径数 + 左子树的路径数 + 右子树的路径数
        return rootSum(root, targetSum) +
               pathSum(root->left, targetSum) +
               pathSum(root->right, targetSum);
    }

    int rootSum(TreeNode* node, long long target) { // 使用long long防止溢出
        if (!node) return 0;
        int count = 0;
        if (node->val == target) count++;
        count += rootSum(node->left, target - node->val);
        count += rootSum(node->right, target - node->val);
        return count;
    }
};
```

# 方法二：前缀和 + 回溯

这个方法利用了前缀和的思想和哈希表进行优化，只需遍历一次二叉树。

核心思想：记录从根节点到当前节点的路径和（前缀和）。如果当前前缀和 curr与某个历史前缀和

prev满足 curr - prev = targetSum，即 curr - targetSum = prev，说明这两个节点之间的路径和就是目标值。

算法步骤：

1. 初始化一个哈希表，记录前缀和及其出现次数，并设置 prefix[0] = 1（空路径和为0）。
2. 深度优先遍历二叉树，更新当前路径和 curr。
3. 检查哈希表中是否存在键 curr - targetSum。若存在，则将对应的次数加入结果。
4. 将当前前缀和 curr加入哈希表，然后递归遍历左右子树。
5. 回溯：在返回当前节点前，将当前前缀和 curr的出现次数减1，避免影响其他分支的统计。

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    unordered_map<long long , int> mp;
    int pathSum(TreeNode* root, int targetSum) {
        if(!root) return 0;
        mp[0] = 1;

        return dfs(root,targetSum,0);
    }

    int dfs(TreeNode* root, int targetSum,long long sj)
    {
        if(!root) return 0;
        int res =0;
        sj += root->val;
        auto si = sj - targetSum;
        if(mp.find(si)!=mp.end())
        {
            res += mp[si];
        }

        mp[sj]++;

        res += dfs(root->right,targetSum,sj);
        res += dfs(root->left,targetSum,sj);
        mp[sj]--;
        return res;
    }
};

class Solution {
    unordered_map<long long, int> prefix; // 键：前缀和，值：出现次数
public:
    int pathSum(TreeNode* root, int targetSum) {
        prefix[0] = 1; // 空路径的基础情况
        return dfs(root, 0, targetSum);
    }

    int dfs(TreeNode* node, long long curr, int target) {
        if (!node) return 0;
        int count = 0;
        curr += node->val; // 更新当前路径和
        // 检查是否存在所需的前缀和
        if (prefix.find(curr - target) != prefix.end()) {
            count += prefix[curr - target];
        }
        prefix[curr]++; // 记录当前前缀和
        count += dfs(node->left, curr, target);
        count += dfs(node->right, curr, target);
        prefix[curr]--; // 回溯，恢复状态
        return count;
    }
};
```
