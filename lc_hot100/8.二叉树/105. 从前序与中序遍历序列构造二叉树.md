/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    // 根据值找到根在中序遍历中的位置
    int find(vector<int>& inorder,int tar)
    {
        for(int i = 0;i<inorder.size();i++)
        {
            if(tar==inorder[i]) return i;
        }
        return -1;
    }
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size();
        if(n==0) return nullptr;
        int tar = preorder[0];

        int index = find(inorder,tar);

        vector<int> p1(preorder.begin() + 1,preorder.begin() + index + 1);
        vector<int> p2(preorder.begin() + index + 1,preorder.end());
        vector<int> i1(inorder.begin(),inorder.begin() + index);
        vector<int> i2(inorder.begin()  + index + 1,inorder.end());

        TreeNode* root = new TreeNode(tar);
        root->left = buildTree(p1,i1);
        root->right = buildTree(p2,i2);

        return root;
    }
};

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        n,m = len(preorder),len(inorder)
        def find(inOrder,k):
            for i in range(len(inOrder)):
                if k == inOrder[i]:
                    return i

            return -1
        if n != m or n == 0 or m == 0:
            return None

        k = find(inorder,preorder[0])
        root = TreeNode(preorder[0])
        p1,p2 = preorder[1:k + 1],preorder[k + 1:]
        i1,i2 = inorder[:k],inorder[k + 1:]
        root.left = self.buildTree(p1,i1)
        root.right = self.buildTree(p2,i2)

        return root
