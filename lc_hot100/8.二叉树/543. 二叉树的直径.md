```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int res = 0;
    int help(TreeNode* root)
    {
        if(!root) return 0;
        auto l = help(root->left);
        auto r = help(root->right);

        res = max(res,l + r);

        return max(l,r) + 1;
    }
    int diameterOfBinaryTree(TreeNode* root) {
        help(root);
        return res;
    }
};
```

关键词 nonlocal 声明当前的变量是外层的变量

```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        res = 0

        def maxDepth(ro) :
            if not ro:
                return 0
            elif not ro.left and not ro.right:
                return 1
            else :
                l = maxDepth(ro.left)
                r = maxDepth(ro.right)
                nonlocal res
                res = max(res,l + r)
                return max(l,r) + 1
        maxDepth(root)
        return res
```
