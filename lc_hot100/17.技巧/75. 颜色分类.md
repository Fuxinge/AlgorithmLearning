```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int n = nums.size();
        int p1 = 0,p2 = 0;
        for(int i = 0;i<n;i++)
        {
            int x = nums[i];
            nums[i] = 2;
            if(x<=1) nums[p1++] = 1;
            if(x==0) nums[p2++] = 0;
        }
    }
};
```
