class Solution {
public:
    void sortColors(vector<int>& nums) {
        int n = nums.size();
        int p1 = 0,p2 = 0;
        for(int i = 0;i<n;i++)
        {
            int x = nums[i];
            nums[i] = 2;
            if(x<=1) nums[p1++] = 1;
            if(x==0) nums[p2++] = 0;
        }
    }
};

class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        a = b = 0
        for i,c in enumerate(nums):
            nums[i] = 2
            if c <= 1:
                nums[a] = 1
                a = a + 1
            if c == 0:
                nums[b] = 0
                b = b + 1
