```cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int res = 0;

        for(auto x:nums)
        {
            res = res^x;
        }

        return res;
    }
};
```

```python
class Solution:
    def singleNumber(self, nums: List[int]) -> int:
        res = 0
        for num in nums:
            res = num ^ res

        return res
```
