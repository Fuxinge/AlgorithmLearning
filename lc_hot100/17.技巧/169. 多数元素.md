class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int dummy = 0,cnt = 0;
        for(auto x:nums)
        {
            if(cnt==0)
            {
                dummy = x;
                cnt++;
            }
            else if(dummy!=x)
            {
                cnt--;
            }
            else cnt++;
        }

        return dummy;
    }
};
