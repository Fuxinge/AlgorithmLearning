必须写成if else的形式

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        auto a = headA,b = headB;

        while(a!= b)
        {
            // 必须写成if else
            if(a == nullptr) a = headB;
            else a = a->next;
            if(b == nullptr) b = headA;
            else b = b->next;
        }

        return a;
    }
};
```
