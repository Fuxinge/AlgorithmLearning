/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if(!head || !head->next) return nullptr;

        auto a = head,b = head;
        while(b&&b->next)
        {
            a = a->next;
            b = b->next->next;

            if(a==b)
            {
                while(b!=head)
                {
                    b = b->next;
                    head = head->next;
                }

                return b;
            }
        }

        return nullptr;
    }
};
