拆分的时候要让后面的多一个

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        ListNode * dummy = nullptr;
        if(!head->next || !head) return true;

        auto a = head,b = head,c = head;

        while(b && b->next)
        {
            c = a;
            a = a->next;
            b = b->next;
            if(b) b = b->next;
        }
        c->next = nullptr;
        b = head;
        auto pre = dummy , cur = a;
        while(cur)
        {
            auto nxt = cur->next;
            cur->next = pre;
            pre = cur;
            cur = nxt;
        }

        while(pre && b)
        {
            if(pre->val != b->val) return false;
            else pre = pre->next,b = b->next;
        }

        return true;
    }
};
```
