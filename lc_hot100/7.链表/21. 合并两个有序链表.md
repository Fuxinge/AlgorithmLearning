/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode * dummy = new ListNode(), *head = dummy;
        auto a = list1,b = list2;
        while(a && b)
        {
            if(a->val <= b->val)
            {
                ListNode * A = new ListNode(a->val);
                a = a->next;
                head->next = A;
                head = head->next;
            }
            else
            {
                ListNode * B = new ListNode(b->val);
                b = b->next;
                head->next = B;
                head = head->next;
            }
        }

        if(a) head->next = a;
        if(b) head->next = b;

        return dummy->next;
    }
};
