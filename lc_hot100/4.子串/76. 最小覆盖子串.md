但是有坑点：check函数要用引用传递，不然会超时

```cpp
class Solution {
public:
    unordered_map<char,int> win,tar;
    bool check(unordered_map<char,int> &win,unordered_map<char,int> &tar)
    {
        for(char a = 'a';a<='z';a++)
        {
            if(win[a]<tar[a]) return false;
        }
        for(char a = 'A';a<='Z';a++)
        {
            if(win[a]<tar[a]) return false;
        }

        return true;
    }
    string minWindow(string s, string t) {
        int n = s.size(),m = t.size();

        for(auto c:t)
        {
            tar[c]++;
        }

        int st = -1,len = INT_MAX;

        for(int i = 0,j = 0;i<n;i++)
        {
            win[s[i]]++;
            while(j <= i && check(win,tar))
            {
                if(i - j + 1 < len)
                {
                    st = j;
                    len = i - j + 1;
                }

                win[s[j]]--;
                j++;
            }
        }
        string res = st==-1 ? "" : s.substr(st,len);
        return res;
    }
};
```

```python
        temp = 0x3f3f3f3f

            cnt_s[c] += 1

                if i - j < ed - st:

                cnt_s[s[j]] -= 1
```
