## 单调队列套路

1. 右边入（元素进入队尾，同时维护队列单调性）
2. 左边出（元素离开队首）
3. 记录/维护答案（根据队首）

```cpp
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int> res;
        int n = nums.size();
        deque<int> q;
        for(int i = 0;i<n;i++)
        {
            // 右入
            while(q.size() && nums[q.back()] <= nums[i])
            {
                q.pop_back();
            }
            q.push_back(i);

            if(i-k+1>q.front()) q.pop_front();

            if(i-k+1>=0) res.push_back(nums[q.front()]);
        }

        return res;
    }
};
```

```python
class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        res = [0] * (n - k + 1)
        q = deque()
        for i,num in enumerate(nums):

            while q and nums[q[-1]] < num:
                q.pop()

            q.append(i)

            left = i - k + 1
            if left < 0:
                continue
            if left > q[0]:
                q.popleft()

            res[left] = nums[q[0]]

        return res

```
