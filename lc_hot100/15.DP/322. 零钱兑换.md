class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int n = coins.size();
        vector<int> dp(amount+1,0x3f3f3f3f);
        dp[0] = 0;
        for(int i = 1;i<=amount;i++)
        {
            for(auto x: coins)
            {
                if(i>=x)
                {
                    dp[i] = min(dp[i],dp[i-x] + 1);
                }
            }
        }

        return dp[amount] > 0x3f3f3f3f/2 ? -1: dp[amount];
    }
};
