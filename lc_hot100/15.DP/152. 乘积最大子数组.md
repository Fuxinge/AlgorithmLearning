class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n = nums.size(),res= INT_MIN;
        vector<int> dp(n + 1,1),udp(n + 1,1);
        for(int i = 0;i<n;i++)
        {
            dp[i+1] = max(dp[i]* nums[i],max(nums[i],nums[i] * udp[i]));
            udp[i+1] = min(dp[i]* nums[i],min(nums[i],nums[i] * udp[i]));
        }

        for(int i = 1;i<=n;i++) res = max(res,dp[i]);

        return res;
    }
};
