```cpp
class Solution {
public:
    int climbStairs(int n) {
        int res = 0;
        int a = 1,b =  2;
        if(n==1) return 1;
        if(n==2) return 2;
        for(int i = 3;i<=n;i++)
        {
            res = a + b;
            a = b;
            b = res;
        }

        return res;
    }
};
```

```python
class Solution:
    def climbStairs(self, n: int) -> int:
        a = 1
        b = 2
        c = 3
        if n == 1:
            return a
        if n ==2:
            return b
        if n==3:
            return c
        for i in range(4,n+1):
            a = b
            b = c
            c = a + b
        return c
```
