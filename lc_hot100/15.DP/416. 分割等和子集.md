```cpp
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int n = nums.size();
        int sum = 0;
        for(auto x: nums) sum+=x;

        int tar = sum/2;
        if(sum%2) return false;

        vector<bool> dp(tar+1,false);
        dp[0] = true;

        for(int i = 0;i<n;i++)
        {
            for(int j = tar;j-nums[i]>=0;j--)
            {
                dp[j] = dp[j] || dp[j-nums[i]];
            }
        }

        return dp[tar];
    }
};
```

```python
class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        s = sum(nums)
        if s%2:
            return False
        s = s // 2

        dp = [False] * (s+1)
        dp[0] = True
        n = len(nums)
        for i in range(n):
            for j in range(s,-1,-1):
                if j >= nums[i]:
                    dp[j] = dp[j] or dp[j - nums[i]]

        return dp[s]
```
