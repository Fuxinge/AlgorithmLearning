class Solution {
public:
    int numSquares(int n) {
        vector<int> can;
        for(int i = 0;i*i<=n;i++)
        {
            can.push_back(i*i);
        }
        if(n==1) return 1;
        if(n==2) return 2;
        vector<int> dp(n+1,INT_MAX/2);
        dp[0] = 0;
        dp[1] = 1,dp[2] = 2;

        for(int i = 3;i<=n;i++)
        {
            for(int j = 1;j*j <= i;j++)
            {
                dp[i] = min(dp[i - j*j] + 1,dp[i]);
            }
        }
        return dp[n];
    }
};

class Solution:
    def numSquares(self, n: int) -> int:
        if n == 1 :
            return 1
        sq = []
        for i in range(1,n+1):
            if i*i <= n:
                sq.append(i*i)
            else :
                break

        dp = [inf] *(n +1)
        dp[0] = 0
        dp[1] = 1

        for i in range(2,n + 1):
            for j in range(len(sq)):
                if sq[j] <= i:
                    dp[i] = min(dp[i],dp[i - sq[j]] + 1)
                else :
                    break

        return dp[n]
