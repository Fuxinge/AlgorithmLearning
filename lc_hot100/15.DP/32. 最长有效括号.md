栈的做法，通俗易懂简单

```cpp
class Solution {
public:
    int longestValidParentheses(string s) {
        int n = s.size();

        stack<int> st;
        st.push(-1);
        int res = 0;
        for(int i = 0;i<n;i++)
        {
            if(s[i] == '(')
            {
                st.push(i);
            }
            else
            {
                st.pop();
                if(st.size()) res = max(res, i - st.top());
                else st.push(i);
            }
        }

        return res;
    }
};
```

```python
class Solution:
    def longestValidParentheses(self, s: str) -> int:
        n = len(s)

        st = [-1]
        res = 0

        for i in range(n):
            if s[i] == '(' :
                st.append(i)
            else :
                st.pop()
                if len(st) :
                    res = max(res, i - st[-1])
                else :
                    st.append(i)
        return res
```
