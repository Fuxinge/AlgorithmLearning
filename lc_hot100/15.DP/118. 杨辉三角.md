```cpp
class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> dp;
        for(int i = 0;i<numRows;i++)
        {
            dp.push_back(vector<int> (i+1,0));
        }

        dp[0][0] = 1;
        if(numRows == 1) return dp;
        dp[1][0] = 1 , dp[1][1] = 1;
        if(numRows == 2) return dp;


        for(int i = 2;i<numRows;i++)
        {
            dp[i][0] = dp[i][i] = 1;
            for(int j = 1;j<i;j++)
            {
                dp[i][j] = dp[i-1][j] + dp[i-1][j-1];
            }
        }

        return dp;
    }
};
```

```python
class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        n = numRows
        dp = [[0] * ( i + 1 ) for i in range(n)]
        dp[0][0] = 1
        if n == 1:
            return dp
        dp[1][1] = dp[1][0] = 1
        if n == 2:
            return dp
        for i in range(2,n):
            dp[i][i] = dp[i][0] = 1
            for j in range(1,i):
                dp[i][j] = dp[i-1][j] + dp[i-1][j-1]

        return dp
```
