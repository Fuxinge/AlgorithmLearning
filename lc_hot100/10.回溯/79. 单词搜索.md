class Solution {
public:
    int dx[4] = {0,1,0,-1},dy[4] = {1,0,-1,0};
    string temp = "";
    bool flag = false;
    void dfs(int x,int y,int u,vector<vector<char>>& board,vector<vector<int>>& st,string word)
    {
        int t = word.size();
        if(u == t)
        {
            flag = true;
            return ;
        }
        int n = board.size(),m = board[0].size();
        for(int i = 0;i<4;i++)
        {
            int px = x + dx[i],py = y + dy[i];
            if(px>=0&&py>=0&&px<n&&py<m&&!st[px][py]&&word[u]==board[px][py])
            {
                st[px][py] = 1;
                dfs(px,py,u+1,board,st,word);
                st[px][py] = 0;
            }
        }
    }
    bool exist(vector<vector<char>>& board, string word) {
        int n = board.size(),m = board[0].size();
        vector<vector<int>> st(n,vector<int>(m,0));
        int t = word.size();
        for(int i = 0;i<n;i++)
        {
            for(int j = 0;j<m;j++)
            {
                if(board[i][j] == word[0])
                {

                    if(t==1) return true;
                    st[i][j] = 1;
                    dfs(i,j,1,board,st,word);
                    st[i][j] = 0;
                    if(flag) return true;
                }
            }
        }
        return false;
    }
};

class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        n, m = len(board), len(board[0])
        size = len(word)

        def dfs(i, j, l):
            if l == size:
                return True
            if j < 0 or j >= m or i < 0 or i >= n or board[i][j] != word[l]:
                return False

            board[i][j] = ''
            for x,y in (i - 1,j),(i,j-1),(i,j+1),(i+1,j):
                if dfs(x,y,l + 1):
                    return True

            board[i][j] = word[l]
            return False

        for i in range(n):
            for j in range(m):
                if dfs(i,j,0):
                    return True

        return False
