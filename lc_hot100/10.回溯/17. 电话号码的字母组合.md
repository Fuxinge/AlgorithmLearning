class Solution {
public:
    vector<string> res;
    unordered_map<char,string> mp;
    void dfs(int u,string digits,string temp)
    {
        if(u==digits.size())
        {
            res.push_back(temp);
            return ;
        }

        string cur = mp[digits[u]];
        for(int i = 0;i<cur.size();i++)
        {
            temp.push_back(cur[i]);
            dfs(u+1,digits,temp);
            temp.pop_back();
        }

        return ;
    }

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        n = len(digits)
        if n == 0:
            return []
        mp = {}
        mp['2'] = "abc"
        mp['3'] = "def"
        mp['4'] = "ghi"
        mp['5'] = "jkl"
        mp['6'] = "mno"
        mp['7'] = "pqrs"
        mp['8'] = "tuv"
        mp['9'] = "wxyz"

        temp = ""
        res = []

        def dfs(i):
            nonlocal res,temp
            if i == n:
                res.append(temp)
                return
            for c in mp[digits[i]]:
                temp += c
                dfs(i+1)
                temp = temp[:-1]

            return

        dfs(0)

        return res
