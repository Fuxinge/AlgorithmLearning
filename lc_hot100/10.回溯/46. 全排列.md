```cpp
class Solution {
public:
    vector<vector<int>> res;
    vector<int> temp,st;
    void dfs(vector<int>& nums,vector<int> & st)
    {
        if(temp.size() == nums.size())
        {
            res.push_back(temp);

            return ;
        }

        for(int i = 0;i<nums.size();i++)
        {
            if(!st[i])
            {
                temp.push_back(nums[i]);
                st[i] = 1;
                dfs(nums,st);
                temp.pop_back();
                st[i] = 0;
            }
        }
    }

    vector<vector<int>> permute(vector<int>& nums) {
        int n = nums.size();
        st.resize(n);

        dfs(nums,st);

        return res;
    }
};
```

```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        n = len(nums)

        res = []
        temp = []
        st = [0] * n
        def dfs(nums,temp,st):
            nonlocal res
            if len(temp) == n:
                res.append(temp[:]) # 这里要复制一份，temp.copy()
                return

            for i,c in enumerate(nums):
                if st[i] == 0:
                    temp.append(c)
                    st[i] = 1
                    dfs(nums,temp,st)
                    st[i] = 0
                    temp.pop()

            return

        dfs(nums,temp,st)

        return res
```
