class Solution {
public:
    vector<vector<int>> res;
    vector<int> temp,st;
    void dfs(vector<int>& nums,vector<int> & st)
    {
        if(temp.size() == nums.size())
        {
            res.push_back(temp);

            return ;
        }

        for(int i = 0;i<nums.size();i++)
        {
            if(!st[i])
            {
                temp.push_back(nums[i]);
                st[i] = 1;
                dfs(nums,st);
                temp.pop_back();
                st[i] = 0;
            }
        }
    }

    vector<vector<int>> permute(vector<int>& nums) {
        int n = nums.size();
        st.resize(n);

        dfs(nums,st);

        return res;
    }
};
