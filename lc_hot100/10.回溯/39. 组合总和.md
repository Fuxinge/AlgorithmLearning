重点是递归的时候第四个参数要用i

```cpp
class Solution {
public:
    vector<vector<int>> res;
    vector<int> temp;
    void dfs(vector<int>& can, int target,int u,int k)
    {
        if(u == target)
        {
            res.push_back(temp);
            return ;
        }
        else if(u>target) return ;

        for(int i = k;i<can.size();i++)
        {
            u += can[i];
            temp.push_back(can[i]);
            dfs(can,target,u,i);
            temp.pop_back();
            u -= can[i];
        }
    }
    vector<vector<int>> combinationSum(vector<int>& can, int target) {
        int n = can.size();
        dfs(can,target,0,0);
        return res;
    }
};
```

```python
class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        n = len(candidates)
        res = []
        temp = []
        def dfs(u,cur ,tar):
            if cur>tar:
                return
            elif cur == tar:
                res.append(temp[:])
                return

            for i in range(u,n):
                cur = cur + candidates[i]
                temp.append(candidates[i])
                dfs(i,cur,tar)
                cur = cur - candidates[i]
                temp.pop()

        dfs(0,0,target)

        return res
```
