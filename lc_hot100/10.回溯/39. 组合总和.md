重点是递归的时候第四个参数要用i

```cpp
class Solution {
public:
    vector<vector<int>> res;
    vector<int> temp;
    void dfs(vector<int>& can, int target,int u,int k)
    {
        if(u == target)
        {
            res.push_back(temp);
            return ;
        }
        else if(u>target) return ;

        for(int i = k;i<can.size();i++)
        {
            u += can[i];
            temp.push_back(can[i]);
            dfs(can,target,u,i);
            temp.pop_back();
            u -= can[i];
        }
    }
    vector<vector<int>> combinationSum(vector<int>& can, int target) {
        int n = can.size();
        dfs(can,target,0,0);
        return res;
    }
};
```
