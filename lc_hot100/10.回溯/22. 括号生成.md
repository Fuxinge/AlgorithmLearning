class Solution {
public:
    vector<string> res;
    string temp;
    unordered_map<char,int> mp;
    void dfs(int n,int u)
    {
        if(u == n * 2)
        {
            res.push_back(temp);
            return ;
        }

        if(mp['('] < n)
        {
            temp = temp + "(";
            mp['('] ++;
            dfs(n,u+1);
            mp['('] --;
            temp.pop_back();
        }

        if(mp['('] > mp[')'])
        {
            temp = temp + ")";
            mp[')'] ++;
            dfs(n,u+1);
            mp[')'] --;
            temp.pop_back();
        }

    }
    vector<string> generateParenthesis(int n) {
        dfs(n,0);
        return res;
    }
};

class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        res = []
        mp = defaultdict(int)
        temp = ''
        def dfs(i) -> None:
            nonlocal temp,res,mp,n
            if i == n * 2:
                res.append(temp)
                return

            if mp['('] < n:
                temp = temp + '('
                mp['('] += 1
                dfs(i + 1)
                mp['('] -= 1
                temp = temp[:-1]

            if mp['('] > mp[')']:
                temp = temp + ')'
                mp[')'] += 1
                dfs(i + 1)
                mp[')'] -= 1
                temp = temp[:-1]

        dfs(0)
        return res
