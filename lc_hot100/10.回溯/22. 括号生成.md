class Solution {
public:
    vector<string> res;
    string temp;
    unordered_map<char,int> mp;
    void dfs(int n,int u)
    {
        if(u == n * 2)
        {
            res.push_back(temp);
            return ;
        }

        if(mp['('] < n)
        {
            temp = temp + "(";
            mp['('] ++;
            dfs(n,u+1);
            mp['('] --;
            temp.pop_back();
        }

        if(mp['('] > mp[')'])
        {
            temp = temp + ")";
            mp[')'] ++;
            dfs(n,u+1);
            mp[')'] --;
            temp.pop_back();
        }

    }
    vector<string> generateParenthesis(int n) {
        dfs(n,0);
        return res;
    }
};
