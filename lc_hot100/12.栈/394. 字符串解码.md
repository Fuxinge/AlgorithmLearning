class Solution {
public:
    string decodeString(string s) {
        int n = s.size();
        string curStr = "",preStr = "";
        int num = 0,preNum = 0;
        stack<int> nums;
        stack<string> strs;

        for(int i = 0;i<n;i++)
        {
            if(s[i] == ']')
            {
                preNum = nums.top();
                preStr = strs.top();
                strs.pop();
                nums.pop();
                string temp = "";
                for(int k = 0;k<preNum;k++)
                {
                    temp = temp + curStr;
                }
                curStr = preStr + temp;
            }
            else if(isdigit(s[i]))
            {
                num = num * 10 + s[i] - '0';
            }
            else if(s[i] == '[')
            {
                nums.push(num);
                num = 0;
                strs.push(curStr);
                curStr = "";
            }
            else
            {
                curStr = curStr + s[i];
            }
        }

        string res = curStr;
        while(strs.size())
        {
            res = strs.top() + res;
            strs.pop();
        }

        return res;
    }
};
