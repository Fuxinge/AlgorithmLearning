class MinStack {
    stack<pair<int,int>> st;
public:
    MinStack() {
        st.push({0,INT_MAX});
    }

    void push(int val) {
        if(val < st.top().second)
        {
            st.push({val,val});
        }
        else
        {
            st.push({val,st.top().second});
        }
    }

    void pop() {
        st.pop();
    }

    int top() {
        return st.top().first;
    }

    int getMin() {
        return st.top().second;
    }
};

class MinStack:

    def __init__(self):
        self.st = deque()
        self.st.append([inf,inf]) # 值和最小值

    def push(self, val: int) -> None:
        mi = min(val,self.st[-1][1])
        self.st.append([val,mi])

    def pop(self) -> None:
        self.st.pop()

    def top(self) -> int:
        return self.st[-1][0]

    def getMin(self) -> int:
        return self.st[-1][1]
