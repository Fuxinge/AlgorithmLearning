class Solution {
public:
    bool isValid(string s) {
        int n = s.size();
        if(n%2==1) return false;
        stack<char> stk;
        for(int i = 0;i<n;i++)
        {
            if(s[i]=='('||s[i]=='['||s[i]=='{')
            {
                stk.push(s[i]);
            }
            else
            {
                if(stk.size()==0) return false;

                if((s[i]==')'&&stk.top()=='(') || (s[i]=='}'&&stk.top()=='{')||(s[i]==']'&&stk.top()=='['))
                {
                    stk.pop();
                }
                else return false;
            }
        }

        return stk.size()==0;
    }
};
