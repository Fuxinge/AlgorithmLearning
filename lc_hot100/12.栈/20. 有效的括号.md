class Solution {
public:
    bool isValid(string s) {
        int n = s.size();
        if(n%2==1) return false;
        stack<char> stk;
        for(int i = 0;i<n;i++)
        {
            if(s[i]=='('||s[i]=='['||s[i]=='{')
            {
                stk.push(s[i]);
            }
            else
            {
                if(stk.size()==0) return false;

                if((s[i]==')'&&stk.top()=='(') || (s[i]=='}'&&stk.top()=='{')||(s[i]==']'&&stk.top()=='['))
                {
                    stk.pop();
                }
                else return false;
            }
        }

        return stk.size()==0;
    }
};

class Solution:
    def isValid(self, s: str) -> bool:
        n = len(s)
        stack = deque()

        for c in s:
            if c == '(' or c == '{' or c == '[':
                stack.append(c)
            elif c == ')':
                if len(stack):
                    t = stack.pop()
                else:
                    return False
                if t != '(':
                    return False
            elif c == ']':
                if len(stack):
                    t = stack.pop()
                else:
                    return False
                if t != '[':
                    return False
            elif c == '}':
                if len(stack):
                    t = stack.pop()
                else:
                    return False
                if t != '{':
                    return False

        return len(stack) == 0
