单调栈:有从左往右和从右往左两种写法

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        int n = t.size();
        stack<int> st; // 一般都是存下标
        vector<int> res(n,0);

        // 从左往右
        for(int i = 0;i<n;i++)
        {
            while(st.size() && t[st.top()] < t[i] ) // 这里似乎有象征意义，< 是找左边第一个小于的，<=就是找小于等于的
            {
                res[st.top()] = i - st.top();
                st.pop();
            }
            st.push(i);
        }
        return res;
    }
};
```

```python
class Solution:
    def dailyTemperatures(self, t: List[int]) -> List[int]:
        n = len(t)
        st = deque()
        res = [0] * n
        for i,c in enumerate(t):
            while len(st) and c > t[st[-1]]:
                idx = st.pop()
                res[idx] = i - idx
            st.append(i)

        return res
```

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        int n = t.size();
        stack<int> st; // 一般都是存下标
        vector<int> res(n,0);

        // 从右往左
        for(int i = n-1;i>=0;i--)
        {
            while(st.size() && t[i] >= t[st.top()])  // 由于栈内的元素意义不一样了，所以上述的象征意义不成立，而是满足条件的都是排除对象 ： 小于等于该元素的不是答案
            {
                st.pop();
            }

            if(st.size()) res[i] = st.top() - i;

            st.push(i);
        }

        return res;
    }
};
```
