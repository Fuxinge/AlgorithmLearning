单调栈:有从左往右和从右往左两种写法

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        int n = t.size();
        stack<int> st; // 一般都是存下标
        vector<int> res(n,0);

        // 从左往右
        for(int i = 0;i<n;i++)
        {
            while(st.size() && t[i] > t[st.top()] )
            {
                res[st.top()] = i - st.top();
                st.pop();
            }
            st.push(i);
        }
        return res;
    }
};
```
