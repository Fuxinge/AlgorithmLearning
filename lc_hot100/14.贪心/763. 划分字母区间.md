class Solution {
public:
    vector<int> partitionLabels(string s) {
        int n = s.size(),prePos = -1;
        vector<int> res;
        unordered_map<char,int> mp;
        for(int i = 0;i<n;i++)
        {
            mp[s[i]] = i;
        }

        int max_right = mp[s[0]];
        for(int i = 0;i<n;i++)
        {
            if(i<max_right)
            {
                max_right = max(max_right,mp[s[i]]);
            }
            if(i == max_right)
            {
                res.push_back(i - prePos);
                prePos = i;
                if(max_right != n-1) max_right = mp[s[i+1]];
            }
        }

        return res;
    }
};
