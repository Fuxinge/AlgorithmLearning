class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        int res = INT_MIN;

        int cur = 0,pre = INT_MAX;
        for(int i = 0;i<n;i++)
        {
            pre = min (pre,prices[i]);
            cur = max(cur,prices[i] - pre);
        }

        return cur;
    }
};

class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        res = 0
        minPrice = inf
        for c in prices:
            minPrice = min(minPrice,c)
            res = max(res,c - minPrice)

        return res
