```cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        int max_right = nums[0];
        for(int i = 0;i<n;i++)
        {
            if(i<=max_right)
            {
                max_right = max(max_right,i + nums[i]);
                if(max_right>=n-1) return true;
            }
        }

        return false;
    }
};
```

```python
class Solution:
    def canJump(self, nums: List[int]) -> bool:
        n = len(nums)
        max_right = 0
        for i,c in enumerate(nums):
            if i<=max_right:
                max_right = max(max_right,i + c)

        return max_right >= n - 1
```
