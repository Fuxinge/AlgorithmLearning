class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        int cur = 0,max_right = nums[0];
        for(int i = 0;i<n;i++)
        {
            if(i<=cur)
            {
                cur = max_right;
                max_right = max(max_right,i + nums[i]);
            }
            if(max_right>=n) return true;
        }

        return false;
    }
};
