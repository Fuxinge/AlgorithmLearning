```cpp
class Solution {
public:
    int jump(vector<int>& nums) {
        int n = nums.size(),max_right = 0,cur = 0,res = 0;

        for(int i = 0;i<n;i++)
        {
            if(i<=cur)
            {
                max_right = max(max_right,i + nums[i]);
            }
            if(i==cur && i!= n-1)
            {
                cur = max_right;
                res++;
            }
        }

        return res;

    }
};
```

```python
class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        max_right = 0
        times = 0
        cur = 0

        for i in range(n - 1):
            max_right = max(max_right,i + nums[i])
            if i == cur:
                times += 1
                cur = max_right

        return times
```
