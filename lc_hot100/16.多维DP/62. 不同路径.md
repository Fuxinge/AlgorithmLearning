```cpp
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> dp(m, vector<int> (n,0));

        dp[0][0] = 1;
        for(int i = 0;i<m;i++)
        {
            for(int j = 0;j<n;j++)
            {
                if(i == 0 || j == 0) dp[i][j] = 1;
                else dp[i][j] = dp[i-1][j] + dp[i][j-1];
            }
        }


        return dp[m-1][n-1];
    }
};
```

```python
class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        total = m + n - 2
        sel = min(m-1,n-1)
        if sel==0:
            return 1
        res = total
        for i in range(1,sel):
            res *= (total - i)

        for i in range(1,sel + 1):
            res = res // i

        return res
```
