```cpp
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int n = grid.size(),m = grid[0].size();

        vector<vector<int> > dp(n+1,vector<int> (m+1,0));

        dp[0][0] = grid[0][0];
        for(int i = 1;i<n;i++)
        {
            dp[i][0] = dp[i-1][0] + grid[i][0];
        }
        for(int j = 1;j<m;j++)
        {
            dp[0][j] = dp[0][j-1] + grid[0][j];
        }

        for(int i = 1;i<n;i++)
        {
            for(int j = 1;j<m;j++)
            {
                dp[i][j] = min(dp[i-1][j],dp[i][j-1]) + grid[i][j];
            }
        }

        for(int i = 0;i<n;i++)
        {
            for(int j = 0;j<m;j++)
            {
                cout<<dp[i][j]<<" ";
            }
            cout<<endl;
        }

        return dp[n-1][m-1];
    }
};
```

```python
class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        n, m = len(grid), len(grid[0])

        # 创建与原网格相同大小的dp数组
        dp = [[0] * m for _ in range(n)]

        # 初始化起点
        dp[0][0] = grid[0][0]

        # 初始化第一行（只能从左向右走）
        for j in range(1, m):
            dp[0][j] = dp[0][j-1] + grid[0][j]

        # 初始化第一列（只能从上向下走）
        for i in range(1, n):
            dp[i][0] = dp[i-1][0] + grid[i][0]

        # 填充其他位置
        for i in range(1, n):
            for j in range(1, m):
                dp[i][j] = min(dp[i-1][j], dp[i][j-1]) + grid[i][j]

        return dp[n-1][m-1]
```
