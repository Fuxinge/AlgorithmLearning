```cpp
class Node {
public:
    Node* son[26] = {};
    bool end = false;
};

class Trie {
    Node* root; // 将 root 声明为 Trie 类的成员变量

    int find(string word) { // 将 find 函数定义为 Trie 类的私有成员函数
        Node* cur = root;
        for (auto c : word) {
            c -= 'a';
            if (!cur->son[c]) {
                return 0;
            }
            cur = cur->son[c];
        }
        return cur->end ? 2 : 1;
    }

public:
    Trie() {
        root = new Node(); // 在构造函数中初始化 root
    }

    void insert(string word) {
        Node* cur = root;
        for (auto c : word) {
            c -= 'a';
            if (!cur->son[c]) {
                cur->son[c] = new Node();
            }
            cur = cur->son[c];
        }
        cur->end = true; // 标记单词的结尾
    }

    bool search(string word) {
        return find(word) == 2;
    }

    bool startsWith(string prefix) {
        return find(prefix) != 0;
    }
};
```
