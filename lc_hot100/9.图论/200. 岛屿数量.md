```cpp
class Solution {
public:
    int dx[4] = {0,1,0,-1},dy[4] = {1,0,-1,0};
    int numIslands(vector<vector<char>>& grid) {
        int n = grid.size(),m = grid[0].size();
        int res = 0;
        vector<vector<int>> st(n,vector<int> (m,0));

        for(int i =0;i<n;i++)
        {
            for(int j = 0;j<m;j++)
            {
                if(grid[i][j] == '1' && st[i][j] == 0)
                {
                    dfs(grid,i,j,st);
                    res++;
                }
            }
        }

        return res;
    }

    void dfs(vector<vector<char>>& grid,int x,int y,vector<vector<int>>& st)
    {
        int n = grid.size(),m = grid[0].size();
        st[x][y] = 1;
        for(int i = 0;i<4;i++)
        {
            int px = x + dx[i],py = y + dy[i];
            if(px>=0 && py>=0 && px<n && py<m &&!st[px][py] && grid[px][py] == '1')
            {
                // st[px][py] = 1;
                dfs(grid,px,py,st);
            }
        }
    }
};
```

```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        n,m = len(grid),len(grid[0])

        # 求联通块的数量 DFS
        def dfs(i: int, j: int) -> None:
            if i < 0 or i >= n or j < 0 or j >= m or grid[i][j] != '1':
                return

            grid[i][j] = '2'
            dfs(i - 1, j)
            dfs(i, j - 1)
            dfs(i + 1, j)
            dfs(i, j + 1)
            return

        res = 0

        for i in range(n):
            for j in range(m):
                if grid[i][j] == '1':
                    dfs(i,j)
                    res += 1

        return res
```
