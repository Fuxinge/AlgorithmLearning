```cpp
class Solution {
public:
    int dx[4] = {0,1,0,-1};
    int dy[4] = {1,0,-1,0};
    int orangesRotting(vector<vector<int>>& grid) {
        int n = grid.size(),m = grid[0].size();
        int cnt = 0;
        queue<pair<int,int> > q;
        for(int i = 0;i<n;i++)
        {
            for(int j = 0;j<m;j++)
            {
                if(grid[i][j] == 1) cnt++;
                else if(grid[i][j]==2) q.push({i,j});
            }
        }
        int time = 0;
        while(q.size())
        {
            if(cnt==0) break;
            int size = q.size();
            for(int i = 0;i<size;i++)
            {
                auto t = q.front();
                q.pop();

                int x = t.first;
                int y = t.second;
                for(int j = 0;j<4;j++)
                {
                    int px = x + dx[j];
                    int py = y + dy[j];
                    if(px>=0&&py>=0&px<n&&py<m&&grid[px][py]==1)
                    {
                        cnt--;
                        grid[px][py] = 2;
                        q.push({px,py});
                    }
                }
            }
            time++;
        }


        return cnt==0?time:-1;
    }
};
```

```python
class Solution:
    def orangesRotting(self, grid: List[List[int]]) -> int:
        n,m = len(grid),len(grid[0])
        cnt = 0
        res = 0
        q = []
        # 传染，BFS
        for i,row in enumerate(grid):
            for j,c in enumerate(row):
                if c == 1:
                    cnt += 1
                elif c == 2:
                    q.append((i,j))


        while q and cnt:
            res += 1
            temp = q
            q = []

            for x,y in temp:
                for i,j in (x - 1, y), (x + 1, y), (x, y - 1), (x, y + 1):
                    if 0<=i<n and 0<=j<m and grid[i][j] == 1:
                        cnt -= 1
                        q.append((i,j))
                        grid[i][j] = 2


        return -1 if cnt else res
```
