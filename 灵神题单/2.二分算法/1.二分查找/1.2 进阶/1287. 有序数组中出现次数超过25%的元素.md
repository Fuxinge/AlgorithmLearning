```python
class Solution:
    def findSpecialInteger(self, arr: List[int]) -> int:
        n = len(arr)
        m = n // 4
        for i in (m,2 * m + 1):
            x = arr[i]
            j = bisect_left(arr,x)
            if arr[j + m] == x:
                return x
        return arr[3 * m + 2]
```
