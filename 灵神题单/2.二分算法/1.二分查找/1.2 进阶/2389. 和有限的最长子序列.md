```python
class Solution:
    def answerQueries(self, nums: List[int], queries: List[int]) -> List[int]:
        def lowerbound(nums,x):
            n = len(nums)
            l,r = 0, n - 1
            while l <= r :
                mid = (l + r) // 2
                if nums[mid] >= x:
                    r = mid - 1
                else :
                    l = mid + 1
            return l

        n,m = len(nums),len(queries)
        res = [0] * m

        nums.sort()
        preSum = [0] * (n + 1)
        for i in range(1,n + 1):
            preSum[i] = preSum[i - 1] + nums[i - 1]

        for i,c in enumerate(queries):
            res[i] = lowerbound(preSum,c + 1) - 1

        return res
```

```python
class Solution:

        for i in range(1, len(nums)):
            nums[i] += nums[i - 1]  # 原地求前缀和
```
