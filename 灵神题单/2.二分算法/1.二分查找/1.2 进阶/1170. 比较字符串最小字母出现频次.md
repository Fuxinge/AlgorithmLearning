```python
class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        n,m = len(queries),len(words)
        res = [0] * n
        wF = [0] * m
        def F(s): # 直接就有On的做法，不排序
            # s = sorted(s)
            # return bisect_right(s,s[0])
            return s.count(min(s))

        for i,c in enumerate(words):
            wF[i] = F(c)
        wF.sort()
        for i,c in enumerate(queries):
            cur = F(c)
            res[i] = m - bisect_left(wF,cur + 1)

        return res
```
