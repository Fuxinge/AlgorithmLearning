class Solution:
    def findTheDistanceValue(self, arr1: List[int], arr2: List[int], d: int) -> int:
        arr2.sort()

        def lowerbound(nums,k):
            n = len(nums)
            l , r = -1 , n
            while l + 1 < r:
                mid = (l + r) // 2
                if nums[mid] >= k:
                    r = mid
                else :
                    l = mid
            return r
        # arr2[j] - d <= arr1[i] <= arr2[j] + d
        res = 0
        for c in arr1:
            pos = lowerbound(arr2 ,c - d)
            if pos == len(arr2) or arr2[pos] > c + d:
                res += 1

        return res
