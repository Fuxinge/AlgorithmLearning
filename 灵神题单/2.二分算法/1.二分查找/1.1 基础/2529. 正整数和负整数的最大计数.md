class Solution:
    def maximumCount(self, nums: List[int]) -> int:
        def lowerbound(nums,k):
            # 找到第一个大于等于k的下标
            n = len(nums)
            l,r = -1,n
            while l + 1 < r:
                mid = (l + r) // 2
                if nums[mid] >= k:
                    r = mid
                else :
                    l = mid
            return r

        pos = len(nums) - lowerbound(nums,1)
        neg = lowerbound(nums,0)
        print(pos,neg)
        return max(neg , pos)
