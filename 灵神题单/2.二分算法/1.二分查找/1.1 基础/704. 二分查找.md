```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        def lowerbound(nums,target):
            n = len(nums)
            i,j = 0, n
            while i < j:
                mid = (i + j) // 2
                if nums[mid] < target:
                    i = mid + 1
                else :
                    j = mid
            return -1 if i == n or nums[i] != target else i

        return lowerbound(nums,target)
```
