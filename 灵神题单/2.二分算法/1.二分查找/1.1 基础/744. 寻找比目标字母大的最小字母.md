class Solution:
    def nextGreatestLetter(self, letters: List[str], target: str) -> str:
        def lowerbound(letters,target):
            n = len(letters)
            l,r = -1, n
            while l + 1 < r:
                mid = (l + r) // 2
                if ord(letters[mid]) <= ord(target):
                    l = mid
                else :
                    r = mid
            if r == n :
                return letters[0]
            else :
                return letters[r]

        return lowerbound(letters,target)
