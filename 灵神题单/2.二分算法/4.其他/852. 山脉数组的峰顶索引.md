class Solution:
    def peakIndexInMountainArray(self, arr: List[int]) -> int:
        n = len(arr)
        l,r = 0,n - 2
        while l + 1 <r:
            mid = (l + r) // 2
            if arr[mid] > arr[mid + 1]:
                r= mid
            else :
                l = mid
        return r

class Solution:
    def peakIndexInMountainArray(self, arr: List[int]) -> int:
        n = len(arr)
        l,r = 1,n - 1
        while l + 1 <r:
            mid = (l + r) // 2
            if arr[mid - 1] > arr[mid]:
                r = mid
            else :
                l = mid
        return l
