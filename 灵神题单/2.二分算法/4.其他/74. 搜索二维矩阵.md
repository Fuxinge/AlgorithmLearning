```python
class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        n,m = len(matrix),len(matrix[0])
        l,r = -1,n*m
        while l + 1 < r:
            mid = (l + r) // 2
            if matrix[mid // m][mid % m] == target:
                return True
            elif matrix[mid // m][mid % m] > target :
                r = mid
            else:
                l = mid

        return False

```
