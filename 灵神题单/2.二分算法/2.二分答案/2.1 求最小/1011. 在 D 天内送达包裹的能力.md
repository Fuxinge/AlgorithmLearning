class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        n = len(weights)
        def check(mid,weights,days):
            cnt = 0
            cur = 0
            for i,c in enumerate(weights):
                if cur + c > mid :
                    cnt += 1
                    cur = c
                else :
                    cur += c
            return cnt + 1 <= days
        l,r = max(weights) - 1, sum(weights)

        while l + 1 < r:
            mid = (l + r) // 2
            if check(mid,weights,days):
                r = mid
            else:
                l = mid
        return r
