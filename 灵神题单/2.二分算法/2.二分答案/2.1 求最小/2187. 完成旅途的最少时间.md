class Solution:
    def minimumTime(self, time: List[int], totalTrips: int) -> int:
        n = len(time)
        l,r = -1 , min(time) * totalTrips + 1
        def check(mid,totalTrips):
            cnt = sum((mid // x) for x in time)
            return cnt >= totalTrips

        while l + 1 < r :
            mid = (l + r) // 2
            if check(mid,totalTrips):
                r = mid
            else :
                l = mid
        return r
