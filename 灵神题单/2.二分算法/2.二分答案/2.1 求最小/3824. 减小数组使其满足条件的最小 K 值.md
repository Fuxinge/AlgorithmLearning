```python
class Solution:
    def minimumK(self, nums: List[int]) -> int:
        def check(nums,mid):
            return sum((x-1) // mid for x in nums) <= mid * mid - len(nums)

        n = len(nums)
        l,r = 0, max(nums)

        while l + 1 < r:
            mid = (l + r) // 2
            if check(nums,mid):
                r = mid
            else:
                l = mid

        return r
```
