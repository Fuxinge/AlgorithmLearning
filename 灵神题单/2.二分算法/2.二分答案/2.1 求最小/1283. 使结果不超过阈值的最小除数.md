```python
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l,r = 0,max(nums)

        while l + 1 < r:
            mid = (l + r) // 2
            if sum((x-1) // mid for x in nums) <= threshold - len(nums):
                r = mid
            else :
                l = mid

        return r
```
