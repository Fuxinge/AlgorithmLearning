class Solution:
    def maxPointsInsideSquare(self, points: List[List[int]], s: str) -> int:
        n = len(points)

        def check(mid,points,s):
            cnt = set()
            for i,item in enumerate(points):
                if max(abs(item[0]),abs(item[1])) <= mid :
                    if s[i] in cnt:
                        return False
                    cnt.add(s[i])

            return True

        l,r = -1,max(max(abs(x[0]),abs(x[1])) for x in points) + 1

        while l + 1 <r:
            mid = (l + r) // 2
            if check(mid,points,s):
                l = mid
            else :
                r = mid
        print(l)
        res = 0
        for c in points:
            if max(abs(c[0]),abs(c[1])) <= l :
                res += 1

        return res
