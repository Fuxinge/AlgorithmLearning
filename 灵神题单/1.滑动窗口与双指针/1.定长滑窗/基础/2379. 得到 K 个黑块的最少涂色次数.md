```python
class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        n,res = len(blocks),inf
        cnt = 0
        for i in range(n):
            if blocks[i] == 'W':
                cnt += 1
            if i < k - 1:
                continue
            res = min(cnt,res)

            if blocks[i-k+1] == 'W':
                cnt -= 1

        return res
```
