```python
class Solution:
    def numOfSubarrays(self, nums: List[int], k: int, threshold: int) -> int:
        n = len(nums)
        res = 0
        threshold = threshold * k
        s = 0
        for i,c in enumerate(nums):
            s += c
            if i < k - 1:
                continue

            if s >= threshold:
                res += 1

            s -= nums[i - k + 1]

        return res
```
