```python
class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        n = len(s)
        cnt = 0
        res = 0
        if n < k:
            return 0
        for i in range(n):
            if s[i] in ('a','e','i','o','u'):
                cnt += 1
            if i < k - 1:
                continue
            res = max(res,cnt)
            if s[i-k+1] in ('a','e','i','o','u'):
                cnt -= 1

        return res

```
