```python
class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        n = len(nums)
        s = 0
        res = -inf
        for i in range(n):
            s += nums[i]
            if i < k- 1:
                continue
            res = max(res,s/k)
            s -= nums[i-k+1]
        return res
```
