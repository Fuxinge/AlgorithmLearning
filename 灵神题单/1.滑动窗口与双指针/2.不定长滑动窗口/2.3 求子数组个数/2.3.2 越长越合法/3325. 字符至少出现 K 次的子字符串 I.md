```python
class Solution:
    def numberOfSubstrings(self, s: str, k: int) -> int:
        n,j,res,mp = len(s),0,0,Counter()

        for i in range(n):
            mp[s[i]] += 1
            while j <= i and mp[s[i]] >= k:
                mp[s[j]] -= 1
                j += 1
            res += j
        return res
```
