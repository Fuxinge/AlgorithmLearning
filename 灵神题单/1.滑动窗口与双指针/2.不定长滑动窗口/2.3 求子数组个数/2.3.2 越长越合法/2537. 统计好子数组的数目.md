class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        n,j,res,mp = len(nums), 0 ,0 ,Counter()
        cnt = 0
        for i in range(n):
            cnt += mp[nums[i]]
            mp[nums[i]] += 1
            while j <= i and cnt >= k:
                mp[nums[j]] -= 1
                cnt -= mp[nums[j]]
                j += 1
            res += j

        return res
