```python
class Solution:
    def validSubstringCount(self, word1: str, word2: str) -> int:
        n,res,j,cnt,mp = len(word1),0,0,defaultdict(int),defaultdict(int)
        for c in word2:
            cnt[c] += 1

        less = len(cnt)

        for i in range(n):
            cnt[word1[i]] -= 1
            if cnt[word1[i]] == 0:
                less -= 1

            while j<=i and less == 0:
                cnt[word1[j]] += 1
                if cnt[word1[j]] == 1:
                    less += 1
                j += 1
            res += j
        return res
```

```python
class Solution:
    def validSubstringCount(self, word1: str, word2: str) -> int:
        n,res,j,cnt,mp = len(word1),0,0,[0]*26,[0]*26
        for c in word2:
            i = ord(c) - ord('a')
            cnt[i] += 1

        def check(win, tar) -> bool:
            """检查窗口是否包含word2的所有字符"""
            for i in range(0,26):
                if win[i] < tar[i]:
                    return False
            return True

        for i in range(n):
            index = ord(word1[i]) - ord('a')
            mp[index] += 1
            while j <=i and check(mp,cnt):
                idx = ord(word1[j]) - ord('a')
                mp[idx] -= 1
                j = j + 1

            res += j
        return res
```
