class Solution:
    def countCompleteSubarrays(self, nums: List[int]) -> int:
        tar = len(Counter(nums))
        n,j,res = len(nums),0,0
        mp = Counter()
        for i in range(n):
            mp[nums[i]] += 1
            while j <= i and len(mp) == tar:
                mp[nums[j]] -= 1
                if mp[nums[j]] == 0:
                    del mp[nums[j]]
                j += 1
            res += j

        return res
