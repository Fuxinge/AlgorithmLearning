```python
class Solution:
    def countSubarrays(self, nums: List[int], k: int) -> int:
        tar = max(nums)
        n,j,res,cnt = len(nums),0,0,0
        for i in range(n):
            if nums[i] == tar:
                cnt += 1
            while j <= i and cnt >= k:
                if nums[j] == tar:
                    cnt -= 1
                j = j + 1
            res += j
        return res
```
