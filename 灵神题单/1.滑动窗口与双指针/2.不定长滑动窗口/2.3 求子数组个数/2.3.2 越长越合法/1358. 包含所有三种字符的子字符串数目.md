```python
class Solution:
    def numberOfSubstrings(self, s: str) -> int:
        n,j,res,mp = len(s),0,0,Counter()

        for i in range(n):
            mp[s[i]] += 1
            while j <= i and mp['a'] > 0 and mp['b'] > 0 and mp['c'] > 0:
                mp[s[j]] -= 1
                j = j + 1

            res += j
        return res
```
