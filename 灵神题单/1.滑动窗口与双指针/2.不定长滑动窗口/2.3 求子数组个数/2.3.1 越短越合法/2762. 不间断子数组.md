```python
class Solution:
    def continuousSubarrays(self, nums: List[int]) -> int:
        n,res,j = len(nums),0,0
        cnt = Counter()
        for i in range(n):
            cnt[nums[i]] += 1
            while j <= i and max(cnt) - min(cnt) > 2:
                cnt[nums[j]] -= 1
                if cnt[nums[j]] == 0 :
                    del cnt[nums[j]]
                j = j + 1

            res += i - j  + 1
        return res
```
