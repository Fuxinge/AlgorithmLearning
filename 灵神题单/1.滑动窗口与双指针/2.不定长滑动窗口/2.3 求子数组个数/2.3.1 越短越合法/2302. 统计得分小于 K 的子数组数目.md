```python
class Solution:
    def countSubarrays(self, nums: List[int], k: int) -> int:
        cur = 0
        res = 0
        n = len(nums)
        j = 0
        for i in range(n):
            cur += nums[i]
            while j <= i and cur*(i - j +1) >= k:
                cur -= nums[j]
                j = j + 1

            res += i - j  + 1
        return res
```
