```python
class Solution:
    def countKConstraintSubstrings(self, s: str, k: int) -> int:
        n = len(s)
        j = 0
        res = 0
        cnt = [0] * 2
        for i in range(n):
            cnt[int(s[i])] += 1
            while j <= n and cnt[0] > k and cnt[1] > k :
                cnt[int(s[j])] -= 1
                j += 1

            res += i - j + 1

        return res
```
