```python
class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        n = len(nums)
        cur = 1
        res = 0
        j = 0
        for i in range(n):
            cur *= nums[i]
            while j <= i and cur >= k:
                cur = cur // nums[j]
                j = j + 1
            res += i - j + 1

        return res
```
