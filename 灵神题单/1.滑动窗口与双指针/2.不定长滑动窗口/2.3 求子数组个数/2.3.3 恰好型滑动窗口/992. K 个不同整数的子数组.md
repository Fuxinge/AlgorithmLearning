class Solution:
    def subarraysWithKDistinct(self, nums: List[int], k: int) -> int:
        def solve(nums,k):
            n,res,j,mp = len(nums),0,0,defaultdict(int)
            for i in range(n):
                mp[nums[i]] += 1
                while j <= i and len(mp)>=k:
                    mp[nums[j]] -= 1
                    if mp[nums[j]] == 0:
                        del mp[nums[j]]
                    j = j +1
                res +=j
            return res

        return solve(nums,k) - solve(nums,k +1)
