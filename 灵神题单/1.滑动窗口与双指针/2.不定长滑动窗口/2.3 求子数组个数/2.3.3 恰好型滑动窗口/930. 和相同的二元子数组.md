class Solution:
    def numSubarraysWithSum(self, nums: List[int], goal: int) -> int:

        def solve(nums,k):
            n,j,res,cur = len(nums),0,0,0
            for i in range(n):
                cur += nums[i]
                while j <= i and cur >= k:
                    cur -= nums[j]
                    j = j + 1
                res += j
            return res

        return solve(nums,goal) - solve(nums,goal + 1)
