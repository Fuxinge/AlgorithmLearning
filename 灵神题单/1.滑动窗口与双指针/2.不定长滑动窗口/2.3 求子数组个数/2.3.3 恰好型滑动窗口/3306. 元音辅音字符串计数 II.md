class Solution:
    def countOfSubstrings(self, word: str, k: int) -> int:
        def solve(word,k):
            n,res,j,cnt = len(word),0,0,Counter()
            c = 0
            for i in range(n):
                cnt[word[i]] += 1
                if word[i] not in ('a','e','i','o','u'):
                    c += 1
                while j <= i and cnt['a'] >= 1 and cnt['e'] >= 1 and cnt['i'] >= 1 and cnt['o'] >= 1 and cnt['u'] >= 1 and c >= k :
                    cnt[word[j]] -= 1
                    if cnt[word[j]] == 0:
                        del cnt[word[j]]
                    if word[j] not in ('a','e','i','o','u'):
                        c -= 1
                    j = j +1
                res +=j
            return res

        return solve(word,k) - solve(word,k +1)
