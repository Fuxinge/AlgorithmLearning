class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        def solve(nums,k):
            n,res,j,cnt = len(nums),0,0,0
            for i in range(n):
                if nums[i] % 2 :
                    cnt += 1
                while j <= i and cnt>=k:
                    if nums[j]%2:
                        cnt -= 1
                    j = j +1
                res +=j
            return res

        return solve(nums,k) - solve(nums,k +1)
