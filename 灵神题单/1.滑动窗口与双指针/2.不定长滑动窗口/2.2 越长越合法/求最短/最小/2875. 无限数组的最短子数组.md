class Solution:
    def minSizeSubarray(self, nums: List[int], target: int) -> int:
        res,j,n = inf,0,len(nums)
        s = sum(nums)
        prefix = 0
        if target == s or target % s == 0:
            return target // s * n
        elif target > s:
            prefix = target // s * n
            target = target % s

        cur = 0
        for i in range(2 * n):
            cur += nums[i % n]
            while j<=i and cur > target:
                cur -= nums[j%n]
                j += 1

            if cur == target:
                res = min(res,i - j +1)

        print(prefix)
        print(res)
        return res + prefix if res!=inf else -1
