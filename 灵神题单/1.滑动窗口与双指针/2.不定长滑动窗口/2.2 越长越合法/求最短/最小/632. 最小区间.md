class Solution:
    def smallestRange(self, nums: List[List[int]]) -> List[int]:
        pairs = sorted((x,i) for (i,arr) in enumerate(nums) for x in arr)
        n = len(nums)
        empty = n
        cnt = [0] * n
        j = 0
        ans_l,ans_r = -inf,inf
        for r,i in pairs:
            cnt[i] += 1
            if cnt[i] == 1:
                empty -= 1

            while empty == 0:
                l,k = pairs[j]
                if r - l < ans_r - ans_l:
                    ans_l,ans_r = l,r
                cnt[k] -= 1
                if cnt[k] == 0:
                    empty += 1
                j += 1

        return [ans_l,ans_r]
