class Solution:
    def minWindow(self, s: str, t: str) -> str:
        n = len(s)
        cnt_t = Counter(t)
        cnt_s = Counter()

        j = 0
        st = -1
        res = inf
        for i in range(n):
            cnt_s[s[i]] += 1
            while j <= i and cnt_s >= cnt_t:
                if res > i - j  + 1:
                    st = j
                    res = i - j + 1

                cnt_s[s[j]] -= 1
                j += 1
        return "" if st < 0 else s[st:st + res]
