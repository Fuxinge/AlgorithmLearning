class Solution:
    def minLength(self, nums: List[int], k: int) -> int:
        n,res,j,cur = len(nums),inf,0,0
        mp = defaultdict(int)

        for i in range(n):
            mp[nums[i]] += 1
            if mp[nums[i]] == 1:
                cur += nums[i]

            while j <= i and cur >= k:
                res = min(res,i - j + 1)
                mp[nums[j ]] -= 1
                if mp[nums[j]] == 0:
                    cur -= nums[j]
                j += 1


        return res if res <= n else -1
