class Solution:
    def balancedString(self, s: str) -> int:
        res = inf
        n = len(s)
        m = n // 4
        mp = Counter(s)
        if  mp['Q'] == m and mp['W'] == m and mp['E'] == m and mp['R'] == m:
            return 0
        j = 0
        for i in range(len(s)):
            mp[s[i]] -= 1
            while j <= i and mp['Q'] <= m and mp['W'] <= m and mp['E'] <= m and mp['R'] <= m:
                res = min(res,i - j + 1)
                mp[s[j]] += 1
                j += 1

        return res if res <= n else 0
