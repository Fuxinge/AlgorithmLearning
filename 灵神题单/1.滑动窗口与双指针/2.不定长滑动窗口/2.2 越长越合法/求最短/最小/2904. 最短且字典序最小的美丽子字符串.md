class Solution:
    def shortestBeautifulSubstring(self, s: str, k: int) -> str:
        res = inf
        n , j , mp = len(s),0,defaultdict(int)
        st = "1" * 1000
        for i in range(n):
            mp[s[i]] += 1
            while j <= i and mp['1'] == k:
                res = min(res,i - j + 1)
                if res < len(st):
                    st = s[j:i+ 1]
                elif res == len(st):
                    st = min(st,s[j:i+1])
                mp[s[j]] -= 1
                j += 1

        return st if len(st) <= n else ""
