```python
class Solution:
    def maximumLengthSubstring(self, s: str) -> int:
        n = len(s)

        mp = defaultdict(int)
        j = 0
        res = 0
        for i in range(n):
            mp[s[i]] += 1
            while j < i and mp[s[i]] > 2:
                mp[s[j]] -= 1
                j += 1
            res = max(res,i - j + 1)

        return res

```
