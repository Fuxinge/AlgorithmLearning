```python
class Solution:
    def maxConsecutiveAnswers(self, answerKey: str, k: int) -> int:
        n = len(answerKey)
        res = 0
        j = 0
        mp = defaultdict(int)

        for i in range(n):
            mp[answerKey[i]] += 1
            while j <= i and min(mp['T'],mp['F']) > k:
                mp[answerKey[j]] -= 1
                j += 1

            res = max(res,i - j + 1)

        return res
```
