class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        n,j,res = len(nums),0,0
        mp = defaultdict(int)
        s = 0
        for i in range(n):
            mp[nums[i]] += 1
            s += nums[i]
            while j <= i and mp[nums[i]] >= 2:
                mp[nums[j]] -= 1
                s -= nums[j]
                j += 1
            res = max(res,s)

        return res
