```python
class Solution:
    def totalFruit(self, fruits: List[int]) -> int:
        mp = defaultdict(int)
        res = 0
        cnt = 0
        j = 0
        for i in range(len(fruits)):
            mp[fruits[i]] += 1
            cnt += 1
            while j <= i and len(mp) > 2:
                mp[fruits[j]] -= 1
                cnt -= 1
                if mp[fruits[j]] == 0:
                    del mp[fruits[j]]
                j += 1
            res = max(res,cnt)

        return res
```
