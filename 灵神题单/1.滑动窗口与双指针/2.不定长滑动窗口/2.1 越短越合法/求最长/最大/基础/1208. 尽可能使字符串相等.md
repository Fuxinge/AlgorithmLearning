```python
class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        res = 0
        cnt = 0
        l = 0
        for r,x in enumerate(s):
            cnt += abs(ord(s[r]) - ord(t[r]))
            while l<=r and cnt > maxCost:
                cnt -= abs(ord(s[l]) - ord(t[l]))
                l += 1
            res = max(res,r - l + 1)

        return res
```
