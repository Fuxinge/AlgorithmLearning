```python
class Solution:
    def minRemoval(self, nums: List[int], k: int) -> int:
        n = len(nums)
        j = 0
        res = -inf
        nums.sort()
        for i in range(n):
            while j <= i and nums[j] * k < nums[i]:
                j += 1
            res = max(res, i - j + 1)

        return n - res

```
