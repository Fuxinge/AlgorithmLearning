```python
class Solution:
    def maxSubarrayLength(self, nums: List[int], k: int) -> int:
        n,j,res = len(nums),0, 0
        mp = defaultdict(int)

        for i in range(n):
            mp[nums[i]] += 1
            while j <= i and mp[nums[i]] > k:
                mp[nums[j]] -= 1
                j += 1
            res = max(res,i - j + 1)

        return res
```
