class Solution:
    def canMakeSubsequence(self, str1: str, str2: str) -> bool:
        def check (s,t)->bool :
            # 判断s 是不是t的子序列
            n = len(s)
            if n == 0:
                return True
            i = 0
            for c in t:
                if c == s[i] or c == chr(ord(s[i]) - 1) or (c == 'z' and s[i] == 'a'):
                    i += 1
                    if i == n :
                        return True

            return False

        return check(str2,str1)
