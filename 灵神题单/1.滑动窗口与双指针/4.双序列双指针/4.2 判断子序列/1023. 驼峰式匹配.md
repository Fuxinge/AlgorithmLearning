class Solution:
    def camelMatch(self, queries: List[str], pattern: str) -> List[bool]:
        n = len(pattern)
        def check(p,s):
            # p 需要是 s 的子串
            n,m = len(p),len(s)
            i = j = 0
            while j < m:
                if i < n and j < m and p[i] == s[j]:
                    i += 1
                    j += 1
                elif s[j].islower():
                    j += 1
                else :
                    return False
            return i == n

        res = []
        for c in queries:
            if check(pattern,c):
                res.append(True)
            else :
                res.append(False)

        return res
