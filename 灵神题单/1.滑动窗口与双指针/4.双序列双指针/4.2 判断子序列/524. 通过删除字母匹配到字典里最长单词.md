```python
class Solution:
    def findLongestWord(self, s: str, dictionary: List[str]) -> str:
        res = ""
        def check (s,t )->bool :
            # 判断s 是不是t的子序列
            n = len(s)
            if n == 0:
                return True
            i = 0
            for c in t:
                if c == s[i]:
                    i += 1
                    if i == n :
                        return True

            return False

        for st in dictionary:
            if check(st,s):
                res = st if len(st) > len(res) or (len(st) == len(res) and st < res) else res

        return res
```
