class Solution:
    def findLUSlength(self, strs: List[str]) -> int:
        n = len(strs)

        def check(s, t) -> bool:
            # 判断s是不是t的子串
            n = len(s)
            if n == 0:
                return True
            if n > len(t):
                return False
            i = 0
            for c in t:
                if c == s[i]:
                    i += 1
                    if i == n :
                        return True

            return False
        res = -1
        for i in range(n):
            s = strs[i]
            flag = True
            for j in range(n):
                if i == j :
                    continue
                t = strs[j]
                if check(s,t):
                    flag = False
                    break
            if flag:
                res = max(len(s),res)

        return res
