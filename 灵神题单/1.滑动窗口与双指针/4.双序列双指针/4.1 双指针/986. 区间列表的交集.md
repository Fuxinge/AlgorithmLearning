class Solution:
    def intervalIntersection(self, firstList: List[List[int]], secondList: List[List[int]]) -> List[List[int]]:
        res = []
        n,m = len(firstList),len(secondList)

        i,j = 0,0
        while i < n and j < m:
            st1,ed1 = firstList[i][0],firstList[i][1]
            st2,ed2 = secondList[j][0],secondList[j][1]

            st = max(st1,st2)
            ed = min(ed1,ed2)

            if st <= ed:
                res.append([st,ed])

            if ed1 < ed2:
                i += 1
            else :
                j += 1

        return res
