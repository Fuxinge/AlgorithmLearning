# 这个题的坑点简直逆天了

```python
class Solution:
    def isLongPressedName(self, name: str, typed: str) -> bool:
        n,m = len(name),len(typed)
        i,j = 0,0

        if name[0] != typed[0]:
            return False

        while i < n and j < m:
            if name[i] == typed[j]:
                i += 1
                j += 1
            elif typed[j] == name[i - 1]:
                j += 1
            else :
                return False


        while j < m :
            if typed[j] != name[-1]:
                return False
            j += 1

        return i == n
```
