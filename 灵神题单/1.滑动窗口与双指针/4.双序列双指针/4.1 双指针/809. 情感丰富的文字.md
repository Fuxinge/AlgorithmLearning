class Solution:
    def expressiveWords(self, s: str, words: List[str]) -> int:
        def check(s,t) -> bool:
            n,m = len(s),len(t)
            if m > n:
                return False

            i,j = 0 , 0
            while i <n and j < m:
                if s[i] != t[j] :
                    return False
                cs = 1
                while i + 1 < n and s[i] == s[i+1]:
                    i += 1
                    cs += 1
                ct = 1
                while j + 1 < m and t[j] == t[j+1]:
                    j += 1
                    ct += 1
                if cs < ct or (cs < 3 and cs != ct):
                    return False

                i += 1
                j += 1

            return i == n and j == m

        res = 0
        for c in words:
            if check(s,c):
                res += 1

        return res
