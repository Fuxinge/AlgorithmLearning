class Solution:
    def canChange(self, s: str, t: str) -> bool:
        n = len(s)
        i = j = 0
        while i < n and j < n:
            while i < n and s[i] == '_':
                i += 1
            while j < n and t[j] == '_':
                j += 1
            if i >= n or j >= n:
                break
            if s[i] != t[j]:
                return False
            else :
                if s[i] == 'L' and i < j:
                    return False
                elif s[i] == 'R' and  i > j:
                    return False
                i += 1
                j += 1

        while i < n :
            if s[i] != '_':
                return False
            i += 1
        while j < n:
            if t[j] != '_':
                return False
            j += 1

        return True
