class Solution:
    def canTransform(self, s: str,t: str) -> bool:
        n = len(s)
        i,j = 0, 0
        while i < n and j < n :
            while i < n and s[i] == 'X':
                i += 1
            while j < n and t[j] == 'X':
                j += 1
            if i >= n or j >= n:
                break
            if s[i] == t[j]:
                if s[i] == 'L' and i < j:
                    return False
                elif s[i] == 'R' and i > j:
                    return False
                else:
                    i += 1
                    j += 1
            else :
                return False

        while i < n :
            if s[i] != 'X':
                return False
            i += 1
        while j < n:
            if t[j] != 'X' :
                return False
            j += 1

        return True
