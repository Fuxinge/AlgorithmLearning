class Solution:
    def smallestDifference(self, a: List[int], b: List[int]) -> int:
        a.sort()
        b.sort()
        n ,m = len(a),len(b)
        i,j = 0 , 0
        res = inf
        while i < n and j < m :
            res = min(res,abs(a[i] - b[j]))
            if a[i] > b[j]:
                j += 1
            elif a[i] == b[j] :
                return 0
            else:
                i += 1

        return res
