class Solution:
    def maxSum(self, nums1: List[int], nums2: List[int]) -> int:
        n,m = len(nums1),len(nums2)
        i,j = 0  , 0
        res = 0
        mod = int(1e9 + 7)
        r1 = r2 = 0
        while i < n or j < m :
            if i < n and (j >= m or nums1[i] < nums2[j]):
                r1 += nums1[i]
                i += 1
            elif j < m and (i >= n or nums1[i] > nums2[j]):
                r2 += nums2[j]
                j += 1
            else :
                r1 = r2 = max(r1,r2) + nums1[i]
                i += 1
                j += 1

        return int(max(r1,r2) % mod)
