```python
class Solution:
    def findKDistantIndices(self, nums: List[int], key: int, k: int) -> List[int]:
        n = len(nums)
        j = 0
        res = []
        for i in range(n):
            if nums[i] != key:
                continue
            j = max(j,i - k)
            while j <= min(i + k,n - 1):
                res.append(j)
                j += 1
        return res
```
