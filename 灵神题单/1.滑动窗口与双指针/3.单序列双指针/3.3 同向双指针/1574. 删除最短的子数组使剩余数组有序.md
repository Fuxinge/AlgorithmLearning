class Solution:
    def findLengthOfShortestSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        right = n - 1
        while right and nums[right] >= nums[right - 1]:
            right -= 1
        if not right:
            return 0

        res = right
        left = 0
        while left == 0 or nums[left] >= nums[left - 1]:
            while right < n and nums[right] < nums[left]:
                right += 1
            res = min(res,right - left - 1)
            left += 1

        return res
