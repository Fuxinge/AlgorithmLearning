```python
class Solution:
    def perfectPairs(self, nums: List[int]) -> int:
        n = len(nums)
        nums.sort(key = abs) #这里要按绝对值排序
        j = 0
        res = 0
        for i in range(n):
            while abs(nums[j]) * 2 < abs(nums[i]):
                j += 1
            res += i - j
        return res
```
