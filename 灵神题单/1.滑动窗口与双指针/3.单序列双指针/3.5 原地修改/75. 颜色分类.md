```python
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        t1 = t2 = 0
        for i,c in enumerate(nums):
            nums[i] = 2
            if c <= 1:
                nums[t1] = 1
                t1 += 1
            if c <= 0:
                nums[t2] = 0
                t2 += 1
```
