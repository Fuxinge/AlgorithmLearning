```python
class Solution:
    def duplicateZeros(self, nums: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        n = len(nums)
        cnt = nums.count(0)
        i,j = n - 1,n + cnt - 1
        while i >= 0:
            if j < n:
                nums[j] = nums[i]
            j -= 1

            if nums[i] == 0:
                if j < n:
                    nums[j] = 0
                j -= 1

            i -= 1
```
