```python
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        n = len(nums)
        i = -1
        j = 0
        while j < n :
            if i == -1 or nums[i] != nums[j]:
                i += 1
                nums[i] = nums[j]

            j += 1
        return i + 1
```
