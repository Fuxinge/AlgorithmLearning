class Solution:
    def isPalindrome(self, s: str) -> bool:

        t = ""
        for c in s:
            if c in string.ascii_lowercase or c in string.ascii_uppercase or c in '1234567890':
                t = t + c
        s = t
        s = s.lower()
        print(s)
        n = len(s)
        i,j = 0, n -1
        while i < j :
            if s[i] != s[j]:
                return False
            i = i + 1
            j =  j - 1

        return True
