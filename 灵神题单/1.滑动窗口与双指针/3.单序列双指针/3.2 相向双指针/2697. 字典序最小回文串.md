class Solution:
    def makeSmallestPalindrome(self, s: str) -> str:
        s = list(s)
        n = len(s)
        i,j = 0, n - 1
        while i < j :
            if s[i] != s[j] :
                if ord(s[i]) < ord(s[j]):
                    s[j] = s[i]
                else :
                    s[i] = s[j]
            i = i + 1
            j = j - 1

        return ''.join(s)
