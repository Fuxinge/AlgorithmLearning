```python
class Solution:
    def triangleNumber(self, nums: List[int]) -> int:
        n = len(nums)
        res = 0
        nums.sort()
        for i in range(n-1,1,-1):
            j,k = 0, i - 1
            while j < k :
                if nums[k] + nums[j] > nums[i]:
                    res += k - j
                    k -= 1
                else:
                    j += 1
        return res
```
