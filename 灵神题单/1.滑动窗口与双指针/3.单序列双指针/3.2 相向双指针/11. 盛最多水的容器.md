```python
class Solution:
    def maxArea(self, nums: List[int]) -> int:
        n = len(nums)
        res = 0
        i = 0
        j = n - 1
        while i < j:
            res = max(res,(j - i) * min(nums[i],nums[j]))
            if nums[i] < nums[j] :
                i += 1
            else :
                j -= 1

        return res
```
