class Solution:
    def bagOfTokensScore(self, nums: List[int], power: int) -> int:
        n = len(nums)
        nums.sort()

        i,j = 0, n - 1
        res = 0
        t = 0
        while i <= j:
            if power >= nums[i]:
                power -= nums[i]
                i += 1
                t += 1
            elif t > 0 and power < nums[i]:
                res = max(res,t)
                power += nums[j]
                j -= 1
                t -= 1
            else :
                break
        return max(res,t)


        return res
