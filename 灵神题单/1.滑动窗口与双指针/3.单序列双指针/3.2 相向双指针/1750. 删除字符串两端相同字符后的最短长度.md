```python
class Solution:
    def minimumLength(self, s: str) -> int:
        n = len(s)
        i,j = 0,n - 1
        while i < j:
            if s[i] != s[j]:
                break
            tar = s[i]
            while i <= j and s[i] == tar:
                i += 1
            while i <= j and s[j] == tar :
                j -= 1
        return j - i + 1
```
