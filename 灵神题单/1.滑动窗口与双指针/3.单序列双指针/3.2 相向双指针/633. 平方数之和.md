```python
class Solution:
    def judgeSquareSum(self, c: int) -> bool:
        a = 0
        b = int(sqrt(c))
        while a*a <= c//2:
            if a*a + b *b == c:
                return True
            elif a*a + b *b < c:
                a = a + 1
            else :
                b = b - 1
        return False
```
