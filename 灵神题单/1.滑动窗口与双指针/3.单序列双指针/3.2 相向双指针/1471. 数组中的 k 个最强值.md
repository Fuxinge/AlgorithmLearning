class Solution:
    def getStrongest(self, arr: List[int], k: int) -> List[int]:
        arr.sort()
        n = len(arr)
        i,j = 0,n - 1
        m = arr[(n-1)//2]

        res = []

        while i <= j and len(res)<k :
            if abs(arr[i] - m) > abs(arr[j] - m) :
                res.append(arr[i])
                i += 1
            else:
                res.append(arr[j])
                j -= 1
        return res
