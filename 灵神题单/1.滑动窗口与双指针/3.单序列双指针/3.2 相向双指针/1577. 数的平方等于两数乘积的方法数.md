class Solution:
    def numTriplets(self, nums1: List[int], nums2: List[int]) -> int:
        def get(nums1,nums2):
            res,n,m = 0,len(nums1),len(nums2)
            nums2.sort()

            for i in range(n):
                j ,k = 0 , m - 1
                while j < k:
                    cur = nums2[j] * nums2[k]
                    tar = nums1[i] ** 2
                    if cur == tar:
                        if nums2[j] == nums2[k]:
                            cnt = k - j +  1
                            res += cnt*(cnt - 1) // 2
                            break # 这里的break很重要
                        else :
                            l,r = nums2[j],nums2[k]
                            t1,t2 = 0,0
                            while nums2[j] == l: # 这里不用写j<k这个条件
                                j += 1
                                t1 += 1
                            while nums2[k] == r: #
                                k -= 1
                                t2 += 1
                            res += t1*t2
                    elif cur < tar:
                        j += 1
                    else :
                        k -= 1

            return res

        return get(nums1,nums2) + get(nums2,nums1)
