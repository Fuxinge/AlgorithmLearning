class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        n = len(nums)
        nums.sort()
        i,j = 0,n - 1
        res = 0
        mod = int(1e9 + 7) # 这里还要加int，不然结果不对
        while i <= j:
            s = nums[i] + nums[j]
            if s <= target:
                le = j - i
                res = (res + 2 ** le ) % mod # 最好还是预处理一下
                i += 1
            else :
                j -= 1

        return int(res % mod)
