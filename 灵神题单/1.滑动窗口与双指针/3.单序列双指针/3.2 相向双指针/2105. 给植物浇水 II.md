```python
class Solution:
    def minimumRefill(self, plants: List[int], ca: int, cb: int) -> int:
        n = len(plants)

        res = 0
        i,j = 0 ,n - 1
        curA,curB = ca,cb
        while i <= j :
            if i == j:
                if curA >= curB and curA < plants[i]:
                    res += 1
                    break

            if curA >= plants[i]:
                curA -= plants[i]
            else :
                res += 1
                curA = ca - plants[i]
            i += 1

            if curB >= plants[j]:
                curB -= plants[j]
            else :
                res += 1
                curB = cb - plants[j]
            j -= 1

        return res
```
