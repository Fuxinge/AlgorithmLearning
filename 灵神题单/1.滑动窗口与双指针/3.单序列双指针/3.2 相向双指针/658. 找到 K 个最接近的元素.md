class Solution:
    def findClosestElements(self, arr: List[int], k: int, x: int) -> List[int]:
        n = len(arr)
        i,j =  0 , n - 1
        while i <= j and j - i + 1 > k:
            if abs(x - arr[i]) <= abs(x- arr[j]):
                j -= 1
            else :
                i += 1

        res = arr[i:j +1]

        return res
