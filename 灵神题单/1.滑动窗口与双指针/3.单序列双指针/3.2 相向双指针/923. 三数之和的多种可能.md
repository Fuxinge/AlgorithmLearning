class Solution:
    def threeSumMulti(self, nums: List[int], target: int) -> int:
        n = len(nums)
        nums.sort()
        res = 0
        mod = int(1e9 + 7)
        for i in range(n- 2):
            j,k = i + 1,n - 1
            while j < k :
                s = nums[i] + nums[k] + nums[j]
                if s == target:
                    if nums[j] != nums[k]:
                        t1,t2 = 0 ,0
                        while j < k and nums[j] == nums[j + 1]:
                            j += 1
                            t1 += 1
                        while j < k and nums[k] == nums[k - 1]:
                            k -= 1
                            t2 += 1
                        j += 1
                        t1 += 1
                        k -= 1
                        t2 += 1
                        res = (res + t1 * t2) % mod
                    else :
                        t = k - j + 1
                        res = (res + t *(t -1) // 2) % mod
                        break
                elif s <target:
                    j += 1
                else :
                    k -= 1
        return res
