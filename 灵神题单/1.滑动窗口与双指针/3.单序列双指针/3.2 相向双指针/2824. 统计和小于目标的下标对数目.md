class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        n = len(nums)
        nums.sort()
        i, j = 0 , n - 1
        res = 0
        print(nums)
        while i<j:
            if nums[i] + nums[j] < target:
                res += j - i
                i = i + 1
            else :
                j = j - 1

        return res
