        n = len(a)
        def check(s)->bool :
            n = len(s)
            left,right = 0,n - 1
            while left < right:
                if s[left] != s[right]:
                    return False
            return True

        for i in range(n + 1):
            aa,ab,ba,bb = a[:i],a[i:],b[:i],b[i:]
            t1 = aa + bb
            t2 = ab + ba
            if check(t1) or check(t2):
                return True
        return False

```python
class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:
        n = len(a)
        def check(a,b)->bool :
            n = len(a)
            left,right = 0,n - 1
            while left < right and a[left] == b[right]:
                left += 1
                right -= 1
            s,t = a[left: right + 1] ,b[left: right + 1]
            return s == s[::-1] or t == t[::-1]

        return check(a,b) or check(b,a)

```
