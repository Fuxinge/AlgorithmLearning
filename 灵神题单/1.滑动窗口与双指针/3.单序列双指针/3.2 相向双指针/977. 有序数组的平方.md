```python
class Solution:
    def sortedSquares(self, nums: List[int]) -> List[int]:
        n = len(nums)
        i,j =  0,  n - 1
        res = []
        while i <= j:
            if abs(nums[i]) > abs(nums[j]):
                res.append(nums[i] ** 2)
                i = i  + 1
            else :
                res.append(nums[j] ** 2)
                j = j - 1
        res.reverse()
        return res
```
