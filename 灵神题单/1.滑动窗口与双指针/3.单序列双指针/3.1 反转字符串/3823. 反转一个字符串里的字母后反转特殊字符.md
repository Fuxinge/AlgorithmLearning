class Solution:
    def reverseByType(self, s: str) -> str:
        st = []
        special = []
        slist = list(s)
        index = [0] * len(s)
        for i in range(len(s)):
            if s[i] in string.ascii_lowercase:
                st.append(s[i])
                index[i] = 1
            else:
                special.append(s[i])

        j = len(st) - 1
        k = len(special) - 1
        for i in range(len(s)):
            if index[i]:
                slist[i] = st[j]
                j -= 1
            else :
                slist[i] = special[k]
                k -= 1


        return ''.join(slist)
