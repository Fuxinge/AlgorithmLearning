```python
class Solution:
    def reverseWords(self, s: str) -> str:
        s = s.split()
        mp = Counter(s[0])
        cnt = mp['a'] + mp['e'] + mp['i'] + mp['o'] + mp['u']

        for i in range(1,len(s)):
            mp = Counter(s[i])
            if cnt == mp['a'] + mp['e'] + mp['i'] + mp['o'] + mp['u']:
                s[i] = ''.join(list(s[i])[::-1])

        return ' '.join(s)

        return ""
```
