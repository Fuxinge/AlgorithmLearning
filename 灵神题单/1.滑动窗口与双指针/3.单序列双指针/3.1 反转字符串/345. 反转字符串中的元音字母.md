class Solution:
    def reverseVowels(self, s: str) -> str:
        s = list(s)
        t = []
        tag = [0] * len(s)
        for i in range(len(s)):
            if s[i] in 'aeiouAEIOU':
                t.append(s[i])
                tag[i] = 1

        j = len(t) - 1
        for i in range(len(s)):
            if s[i] in 'aeiouAEIOU':
                s[i] = t[j]
                j = j - 1

        return ''.join(s)
