```python
class Solution:
    def reverseSubmatrix(self, grid: List[List[int]], x: int, y: int, k: int) -> List[List[int]]:
        left = y
        right = y + k
        up = x
        down = x + k - 1

        while up<down:
            grid[up][left:right],grid[down][left:right] = grid[down][left:right],grid[up][left:right]
            up += 1
            down -= 1
        return grid
```
