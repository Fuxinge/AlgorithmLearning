class Solution:
    def reverseOnlyLetters(self, s: str) -> str:
        s = list(s)
        t = []
        tag = [0] * len(s)
        for i in range(len(s)):
            if s[i] in string.ascii_lowercase or s[i] in string.ascii_uppercase:
                t.append(s[i])
                tag[i] = 1

        j = len(t) - 1
        for i in range(len(s)):
            if s[i] in string.ascii_lowercase or s[i] in string.ascii_uppercase:
                s[i] = t[j]
                j = j - 1

        return ''.join(s)
