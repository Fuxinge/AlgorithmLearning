```python
class Solution:
    def countNegatives(self, grid: List[List[int]]) -> int:
        n,m = len(grid),len(grid[0])
        i,j = 0,m - 1
        res = 0
        while i < n and j >= 0:
            if grid[i][j] < 0:
                res += n - i
                j -= 1
            else :
                i += 1
        return res
```
