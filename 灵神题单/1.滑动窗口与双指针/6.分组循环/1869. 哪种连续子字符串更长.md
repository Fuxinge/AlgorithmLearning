class Solution:
    def checkZeroOnes(self, s: str) -> bool:
        cnt1 = cnt0 = 0
        res1 = res0 = 0
        for c in s:
            if c == '1' :
                cnt0 = 0
                cnt1 += 1
                res1 = max(res1,cnt1)
            else :
                cnt1 = 0
                cnt0 += 1
                res0 = max(res0,cnt0)
        print(res1,res0)
        return res1 > res0
