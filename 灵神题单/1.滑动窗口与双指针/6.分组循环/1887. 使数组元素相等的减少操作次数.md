```python
class Solution:
    def reductionOperations(self, nums: List[int]) -> int:
        res = i = 0
        n = len(nums)
        nums.sort(reverse=True)

        while i < n:
            i += 1
            while i < n and nums[i] == nums[i - 1]:
                i += 1
            if i == n:
                return res
            else :
                res += i
        return res
```
