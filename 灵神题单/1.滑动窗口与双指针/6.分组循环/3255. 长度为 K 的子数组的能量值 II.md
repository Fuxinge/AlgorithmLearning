class Solution:
    def resultsArray(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        res = [-1] * (n - k + 1)
        i = 0
        while i < n :
            s = i
            i += 1
            while i < n and nums[i] == nums[i - 1] + 1:
                i += 1
            l = i - s
            for j in range(s,s + l - k + 1):
                # print(l,j,j+k-1)
                res[j] = nums[j + k - 1]
        return res
