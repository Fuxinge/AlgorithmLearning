```python
class Solution:
    def findScore(self, nums: List[int]) -> int:
        n = len(nums)
        res = 0
        i = 0
        while i < n:
            s = i
            while i + 1 < n and nums[i] > nums[i + 1]:
                i += 1
            for j in range(i,s - 1,-2):
                res += nums[j]
            i += 2
        return res
```
