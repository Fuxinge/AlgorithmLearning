```python
class Solution:
    def longestSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        i = 0
        res = 2
        while i < n :
            start = i
            i += 1
            while 2 <= i < n and nums[i] == nums[i - 1] + nums[i - 2]:
                i += 1
            res = max(res,i - start + 1) # 这里多一位
        return res
```

```python
class Solution:
    def longestSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        i = 0
        res = 2
        while i < n :
            start = i
            i += 2 # 这里要加2
            while 2 <= i < n and nums[i] == nums[i - 1] + nums[i - 2]:
                i += 1
            res = max(res,i - start)
            i -= 1
        return res
```
