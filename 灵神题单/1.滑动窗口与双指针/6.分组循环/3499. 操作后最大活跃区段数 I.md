class Solution:
    def maxActiveSectionsAfterTrade(self, s: str) -> int:
        s = '1' + s + '1'
        n = len(s)
        res = i = mx = 0
        pre = -inf
        cnt = Counter(s)
        while i < n :
            st = i
            i += 1
            while i < n and s[i] == s[i - 1]:
                i += 1
            if s[st] == '0':
                mx = max(mx,pre + i - st)
                pre = i - st
        return mx + cnt['1'] - 2
