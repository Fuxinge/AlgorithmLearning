```python
class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        n = len(nums)
        res = i = 0
        if n < 3 :
            return 0
        def cnt(n:int)->int:
            # 1,1+2,1+2+3
            n -= 2
            return n * (n + 1) // 2

        while i < n :
            s = i
            i += 1
            if i == n:
                break
            d = nums[i] - nums[i - 1]
            i += 1
            while i < n and nums[i] - nums[i - 1] == d:
                i += 1
            if i - s >= 3:
                res += cnt(i-s)

            i -= 1
        return res
```
