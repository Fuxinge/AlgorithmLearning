class Solution:
    def makeFancyString(self, s: str) -> str:
        res = ""
        i = 0
        n = len(s)
        while i < n :
            start = i
            i += 1
            while i < n and s[i] == s[i- 1]:
                i += 1
            if i - start == 1:
                res += s[start]
            else :
                res += s[start]
                res += s[start]
        return res
