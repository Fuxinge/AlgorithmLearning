```python
class Solution:
    def zeroFilledSubarray(self, nums: List[int]) -> int:
        res = i = 0
        n = len(nums)

        while i < n:
            if i > n or nums[i] != 0:
                i += 1
                continue

            start = i
            i += 1

            while i < n and nums[i] == 0:
                i += 1
            if i - start > 0:
                res += (i-start + 1)* (i-start) // 2

        return res
```
