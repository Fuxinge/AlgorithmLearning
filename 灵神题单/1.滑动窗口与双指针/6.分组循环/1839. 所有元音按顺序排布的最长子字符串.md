class Solution:
    def longestBeautifulSubstring(self, word: str) -> int:
        res = i = 0
        n = len(word)
        mp = {}
        mp['a'] = 1
        mp['e'] = 2
        mp['i'] = 3
        mp['o'] = 4
        mp['u'] = 5

        while i < n :
            s = i
            i += 1
            while i < n and word[s] == 'a' and mp[word[i]] >= mp[word[i - 1]]:
                i += 1
            if i - s >= 5 and word[s] == 'a' and  word[i -1] == 'u' and res < i - s:
                print(s,i-1)
                print(word[s:i])
                cnt = Counter(word[s:i])
                if cnt['a'] and cnt['e'] and cnt['i'] and cnt['o'] and cnt['u'] :
                    res = max(res, i -s)

        return res
