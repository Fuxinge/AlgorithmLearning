class Solution:
    def minCost(self, colors: str, neededTime: List[int]) -> int:
        n = len(colors)
        res = i = 0
        ma = 0
        while i < n :
            s = i
            ma = neededTime[i]
            i += 1
            while i < n and colors[i] == colors[i - 1]:
                ma = max(ma,neededTime[i])
                i += 1
            if i - s > 1:
                su = sum(neededTime[s:i])
                res += su - ma
                # print(su,ma,res)
        return res
