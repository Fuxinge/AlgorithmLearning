```python
class Solution:
    def numSub(self, s: str) -> int:
        n = len(s)
        res = i = 0
        mod = int(1e9+ 7)
        while i < n :
            if s[i] == '0' :
                i += 1
                continue
            start = i
            i += 1
            while i < n and s[i] == '1':
                i += 1

            res += (i - start) * (i - start + 1) // 2 % mod
        return res % mod
```
