```python
class Solution:
    def getDescentPeriods(self, prices: List[int]) -> int:
        n = len(prices)
        i = res = 0
        while i < n :
            s = i
            i += 1
            while i < n and prices[i] == prices[i - 1] - 1:
                i += 1
            res += (i-s) *(i - s + 1) // 2

        return res
```
