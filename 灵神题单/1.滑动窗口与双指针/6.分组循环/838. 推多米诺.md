class Solution:
    def pushDominoes(self, dominoes: str) -> str:

        res = list("L" + dominoes + "R")
        i = 0
        dominoes = "L" + dominoes + "R"
        n = len(dominoes)
        while i < n:
            s = i
            i += 1
            while i < n and dominoes[i] == '.':
                i += 1
            if i < n and dominoes[i] == dominoes[s]:
                res[s + 1:i] = dominoes[i]*(i - s - 1)
            elif i < n and  dominoes[s] == 'R':
                half = (i - s - 1) // 2
                res[s + 1:s + half + 1] = 'R' * half
                res[i - half:i] = 'L' * half
        return ''.join(res[1:-1])
