class Solution:
    def longestMonotonicSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        r1 = r2 = i = 0

        while i < n :
            s = i
            i += 1
            while i < n and nums[i] > nums[i - 1]:
                i += 1
            r1 = max(r1,i - s)

        i = 0
        while i < n :
            s = i
            i += 1
            while i < n and nums[i] < nums[i - 1]:
                i += 1
            r1 = max(r1,i - s)

        return r1
