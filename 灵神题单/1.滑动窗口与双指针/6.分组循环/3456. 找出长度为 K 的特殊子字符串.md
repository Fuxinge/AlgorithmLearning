```python
class Solution:
    def hasSpecialSubstring(self, s: str, k: int) -> bool:
        n = len(s)

        i = 0
        while i < n :
            start = i
            i += 1
            while i < n and s[i] == s[i-1]:
                i += 1
            if i - start == k:
                return True

        return False
```
