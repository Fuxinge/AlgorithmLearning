```python
class Solution:
    def findLengthOfLCIS(self, nums: List[int]) -> int:
        n = len(nums)
        i = 0
        res = 0

        while i < n :
            start = i
            i += 1
            while i < n and nums[i] > nums[i-1]:
                i += 1
            res = max(res,i - start)

        return res
```
