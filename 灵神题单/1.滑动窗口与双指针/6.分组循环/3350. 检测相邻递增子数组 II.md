class Solution:
    def maxIncreasingSubarrays(self, nums: List[int]) -> int:
        n = len(nums)
        res = 1
        i = 0
        flag = []

        while i < n :
            s = i
            i += 1
            while i < n and nums[i] > nums[i - 1]:
                i += 1
            flag.append([s,i - 1])
            res = max(res,(i-s)//2)

        for st,ed in pairwise(flag):
            if st[1] + 1 == ed[0]:
                res = max(res,min(ed[1] - ed[0] + 1,st[1] - st[0] + 1))
        return res
