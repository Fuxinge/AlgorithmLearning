```python
class Solution:
    def canSortArray(self, nums: List[int]) -> bool:
        n = len(nums)
        i = 0

        while i < n:
            s = i
            cnt = nums[i].bit_count()
            i += 1
            while i < n and cnt == nums[i].bit_count():
                i += 1
            nums[s:i] = sorted(nums[s:i])

        k = nums[:]
        nums.sort()
        return k == nums # 用循环判断可O(1)空间复杂度
```
