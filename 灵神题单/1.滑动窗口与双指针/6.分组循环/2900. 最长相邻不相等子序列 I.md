```python
class Solution:
    def getLongestSubsequence(self, words: List[str], groups: List[int]) -> List[str]:
        n = len(words)
        res = []
        i = 0

        while i < n :
            s = i
            res.append(words[s])
            i += 1
            while i < n and groups[i] == groups[i - 1]:
                i += 1
        return res
```
