class Solution:
    def longestMountain(self, nums: List[int]) -> int:
        res = i = 0
        n = len(nums)
        if n < 3 :
            return 0
        idx = 0
        while i < n:
            s = i
            i += 1
            while i < n and nums[i] > nums[i - 1]:
                i += 1
            idx = i # idx 记录的是山顶的下一个位置（或者说上坡停止的位置）
            # 如果没有上坡，直接进入下一轮寻找
            if idx == s + 1:
                continue
            while i < n and nums[i] < nums[i - 1]:
                i += 1
            if i > idx :#  必须有下坡才算真正的山脉！(i 必须大于上坡停止的 idx)
                res = max(res,i - s)
                i -= 1
        return res
