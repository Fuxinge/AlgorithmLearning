```python
class Solution:
    def countHomogenous(self, s: str) -> int:
        n = len(s)
        mod = int(1e9 + 7)
        res = i = 0
        while i < n:
            st = i
            i += 1
            while i < n and s[i] == s[i - 1]:
                i += 1
            res = res + (i - st) * (i - st + 1) // 2
            res = int(res % mod)

        return int(res % mod)
```
