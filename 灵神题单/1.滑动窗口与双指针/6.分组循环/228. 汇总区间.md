```python
class Solution:
    def summaryRanges(self, nums: List[int]) -> List[str]:
        res = []
        i = 0
        n = len(nums)
        while i < n :
            s = i
            i += 1
            while i < n and nums[i] == nums[i - 1] + 1:
                i += 1
            if i - s == 1:
                res.append(f"{nums[s]}")
            else :
                res.append(f"{nums[s]}->{nums[s] + i - s - 1}")
        return res
```
