```python
class Solution:
    def countBinarySubstrings(self, s: str) -> int:
        n = len(s)
        i = 0
        res = 0
        pre = 0
        cur = 0
        while i < n:
            start = i
            i += 1
            while i < n and s[i] == s[i - 1]:
                i += 1
            cur = i - start
            res += min(cur,pre)
            pre = cur

        return res
```
