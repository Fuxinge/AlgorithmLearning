class Solution:
    def maxTurbulenceSize(self, nums: List[int]) -> int:
        i = t = 0
        n = len(nums)
        s = 0
        while i < n:
            start = i
            i += 1
            while i < n and (i % 2 and nums[i] > nums[i - 1] or i % 2 == 0 and nums[i] < nums[i - 1]):
                i += 1
            t = max(t,i-start)

        i = 0
        while i < n:
            start = i
            i += 1
            while i < n and (i % 2 and nums[i] < nums[i - 1] or i % 2 == 0 and nums[i] > nums[i - 1]):
                i += 1
            s = max(s,i - start)

        return max(s,t)
