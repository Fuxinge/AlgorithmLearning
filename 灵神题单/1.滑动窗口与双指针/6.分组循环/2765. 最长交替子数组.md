```python
class Solution:
    def alternatingSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1:
            return - 1
        res = -1
        i = 0
        while i < n :
            s = i
            i += 1
            if i == n or nums[i] != nums[i - 1] + 1 :
                continue
            i += 1 # i已经判断过了
            while i < n and nums[i] == nums[i - 2]:
                i += 1
            res = max(res,i - s)
            i -= 1

        return res

```
