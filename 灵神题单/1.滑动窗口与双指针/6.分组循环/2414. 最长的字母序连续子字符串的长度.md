```python
class Solution:
    def longestContinuousSubstring(self, s: str) -> int:
        n = len(s)
        def check(a,b):
            # 判断a的下一个字符是不是b
            return ord(a) == ord(b) - 1

        res = i = 0
        while i < n:
            start = i
            i += 1
            while i < n and check(s[i-1],s[i]):
                i += 1
            res = max(res,i - start)

        return res
```
