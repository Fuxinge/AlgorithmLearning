class NumArray:

    def __init__(self, nums: List[int]):
        self.s = [0] * (len(nums) + 1)
        for i,c in enumerate(nums):
            self.s[i+1] = self.s[i] + c

    def sumRange(self, left: int, right: int) -> int:
        return self.s[right + 1] - self.s[left]
