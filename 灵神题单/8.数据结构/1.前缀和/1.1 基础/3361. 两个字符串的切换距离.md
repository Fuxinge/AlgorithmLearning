```python
class Solution:
    def shiftDistance(self, s: str, t: str, nextCost: List[int], previousCost: List[int]) -> int:
        res = 0
        pre = list(accumulate(previousCost, initial=0))
        nxt = list(accumulate(nextCost, initial=0))

        for cur, tar in zip(s, t):
            if cur == tar:
                continue

            # 拿到字符对应的 0-25 下标
            i = ord(cur) - ord('a')
            j = ord(tar) - ord('a')

            # 1. 独立计算：全速向前走的代价
            if i <= j:
                forward_cost = nxt[j] - nxt[i]
            else:
                forward_cost = nxt[26] - nxt[i] + nxt[j]

            # 2. 独立计算：全速向后退的代价
            if i >= j:
                backward_cost = pre[i + 1] - pre[j + 1]
            else:
                backward_cost = pre[i + 1] + pre[26] - pre[j + 1]

            # 3. 比较两者，取最小！
            res += min(forward_cost, backward_cost)

        return res
```
