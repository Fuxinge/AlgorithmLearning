```python
class Solution:
    def vowelStrings(self, words: List[str], queries: List[List[int]]) -> List[int]:
        ae = ('a','e','i','o','u')
        n = len(words)
        f = 0
        s = [0] * (n + 1)
        for i,w in enumerate(words):
            if w[0] in ae and w[-1] in ae:
                f = 1
            else :
                f = 0
            s[i + 1] = s[i] + f
        res = []
        for l,r in queries:
            res.append(s[r+1] - s[l])

        return res
```
