```python
class Solution:
    def subarraySum(self, nums: List[int]) -> int:
        n = len(nums)
        res = 0
        s = [0] * (n + 1)
        for i,c in enumerate(nums):
            s[i+1] = s[i] + c

        for i in range(len(nums)):
            st = max(0,i - nums[i])
            res += s[i + 1] - s[st]

        return res
```
