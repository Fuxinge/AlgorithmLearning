class Solution:
    def maxProfit(self, prices: List[int], strategy: List[int], k: int) -> int:
        n = len(prices)

        c = [p*c for p,c in zip(prices,strategy)]
        s = list(accumulate(c,initial = 0))
        print(n,len(c),len(s))
        p = list(accumulate(prices,initial = 0))

        # i-k 到 i - 1 闭区间
        res = max(s[i - k] + s[n] - s[i] + p[i] - p[i - k // 2] for i in range(k,n + 1))
        return max(res,s[n])
