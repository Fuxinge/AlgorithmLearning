```python
class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        n = len(arr)
        s = [0] * (n + 1)
        for i,c in enumerate(arr):
            s[i + 1] = s[i] ^ c
        res = []
        for l,r in queries:
            res.append(s[r+1] ^ s[l])

        return res
```
