```python
class Solution:
    def isArraySpecial(self, nums: List[int], queries: List[List[int]]) -> List[bool]:
        n = len(nums)
        s = [0] * (n + 1)
        for i in range(1,n):
            s[i] = s[i - 1] + (nums[i] % 2 == nums[i - 1] % 2)

        res = [ ]
        for l,r in queries:
            res.append(s[l] == s[r])
        return res
```
