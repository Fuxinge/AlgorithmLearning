```python
class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        res = 0
        mp = defaultdict(int)

        s = list(accumulate(nums,initial = 0))
        # sj - si = k
        for sj in s:
            si = sj - k
            if mp[si] != 0:
                res += mp[si]
            mp[sj] += 1
        return res
```
