```python
class Solution:
    def getLargestOutlier(self, nums: List[int]) -> int:
        s = sum(nums)
        res = -inf
        mp = Counter(nums)

        for c in nums:
            mp[c] -= 1
            if (s - c) % 2 == 0 and mp[(s-c)// 2] > 0:
                res = max(res,c)
            mp[c] += 1


        return res


```
