```python
class Solution:
    def maxSum(self, nums: List[int]) -> int:
        mp = defaultdict(int) # 由于都是两位数，可以直接用数组，常数空间复杂度
        res = - 1
        def getMax(num):
            # 学习灵神写法 max(map(int, str(num)))
            return ord(max(str(num))) - ord('0')
        for i,c in enumerate(nums):
            a = getMax(c)
            if a in mp:
                res = max(res,c + mp[a])
            mp[a] = max(mp[a],c)

        return res
```
