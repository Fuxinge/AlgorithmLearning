```python
class Solution:
    def interchangeableRectangles(self, nums: List[List[int]]) -> int:
        res = 0
        mp = defaultdict(int)

        for i,c in enumerate(nums):
            if c[0] / c[1] in mp:
                res += mp[c[0] / c[1]]

            mp[c[0] / c[1]] += 1

        return res
```
