```python
class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        res = 0
        mp = defaultdict(int)
        for i,c in enumerate(nums):
            if k - c in mp and mp[k-c] > 0:
                res += 1
                mp[k-c] -= 1
            else :
                mp[c] += 1

        return res
```
