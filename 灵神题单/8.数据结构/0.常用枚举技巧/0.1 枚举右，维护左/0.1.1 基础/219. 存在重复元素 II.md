```python
class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
        n = len(nums)
        mp = defaultdict(int)
        for i,c in enumerate(nums):
            if c in mp:
                if i - mp[c] <= k:
                    return True
            mp[c] = i

        return False
```
