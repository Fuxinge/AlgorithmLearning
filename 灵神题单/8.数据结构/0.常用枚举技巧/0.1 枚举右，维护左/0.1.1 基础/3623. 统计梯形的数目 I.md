```python
class Solution:
    def countTrapezoids(self, points: List[List[int]]) -> int:
        mod = int(1e9 + 7)

        mp = Counter(p[1] for p in points)
        res = 0
        s = 0
        for c in mp.values():
            k = c * (c - 1) // 2
            res += s * k
            s += k

        return res % mod

```
