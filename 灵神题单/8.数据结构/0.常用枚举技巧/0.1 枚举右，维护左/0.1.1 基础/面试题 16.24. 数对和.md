class Solution:
    def pairSums(self, nums: List[int], target: int) -> List[List[int]]:
        res = []
        mp = defaultdict(int)
        for i,c in enumerate(nums):
            if target - c in mp and mp[target -c ] > 0:
                res.append([c,target - c])
                mp[target - c] -= 1
            else :
                mp[c] += 1

        return res
