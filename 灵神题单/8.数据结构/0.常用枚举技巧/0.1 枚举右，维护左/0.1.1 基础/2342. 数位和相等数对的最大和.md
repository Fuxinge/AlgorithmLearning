```python
class Solution:
    def maximumSum(self, nums: List[int]) -> int:
        mp = defaultdict(int)
        res = -1
        n = len(nums)
        for i,c in enumerate(nums):
            cnt = 0
            x = c
            while x:
                cnt += x % 10
                x //= 10
            if cnt in mp:
                res = max(res,c + mp[cnt])
            mp[cnt] = max(mp[cnt],c)

        return res
```
