```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        mp = defaultdict(int)
        n = len(nums)

        for i,c in enumerate(nums):
            if target - c in mp:
                return [i,mp[target - c ]]

            mp[c] = i

        return [-1,-1]
```
