```python
class Solution:
    def numEquivDominoPairs(self, dominoes: List[List[int]]) -> int:
        n = len(dominoes)
        mp = defaultdict(int)
        res = 0
        for i,c in enumerate(dominoes):
            d = tuple(sorted(c))
            res += mp[d]
            mp[d] += 1

        return res
```
