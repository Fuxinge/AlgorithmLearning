```python
class Solution:
    def countPairs(self, words: List[str]) -> int:
        n,m = len(words),len(words[0])

        mp = defaultdict(int)
        res = 0

        for s in words:
            t = list(s)
            base = ord(t[0])
            for i in range(len(t)):
                t[i] = chr((ord(t[i]) - base) % 26)
            t = ''.join(t)
            res += mp[t]
            mp[t] += 1

        return res
```
