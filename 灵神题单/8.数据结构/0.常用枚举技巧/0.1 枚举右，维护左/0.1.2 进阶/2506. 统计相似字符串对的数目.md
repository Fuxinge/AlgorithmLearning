```python
class Solution:
    def similarPairs(self, words: List[str]) -> int:
        mp = defaultdict(int)
        res = 0
        for s in words:
            cur = [0] * 26
            for c in s:
                cur[ord(c) - ord('a')] = 1
            cur = tuple(cur)
            if mp[cur] != 0:
                res += mp[cur]
            mp[cur] += 1

        return res
```
