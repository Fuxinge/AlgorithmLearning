```python
class Solution:
    def countBeautifulPairs(self, nums: List[int]) -> int:
        res = 0
        mp = [0] * 10
        for c in nums:
            for j,x in enumerate(mp):
                if x and gcd(c % 10,j) == 1:
                    res += x
            t = c
            while t >= 10 :
                t //= 10
            mp[t] += 1

        return res

```
