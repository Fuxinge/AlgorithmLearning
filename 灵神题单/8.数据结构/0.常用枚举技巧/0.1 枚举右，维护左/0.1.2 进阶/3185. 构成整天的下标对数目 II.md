```python
class Solution:
    def countCompleteDayPairs(self, hours: List[int]) -> int:
        n = len(hours)
        mp = defaultdict(int)

        res = 0
        for i,c in enumerate(hours):
            if (24 - c % 24) % 24 in mp:
                res += mp[(24 - c % 24) % 24]
            mp[c%24] += 1

        return res
```
