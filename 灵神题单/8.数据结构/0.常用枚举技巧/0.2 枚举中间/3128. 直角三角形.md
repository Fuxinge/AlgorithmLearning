class Solution:
    def numberOfRightTriangles(self, grid: List[List[int]]) -> int:
        n,m = len(grid),len(grid[0])
        res = 0

        def check(i,j,n,m):
            return i >= 0 and j >= 0 and i < n and j < m

        col = [-1] * m
        row = [-1] * n
        for i in range(n):
            for j in range(m):
                if grid[i][j]:
                    row[i] += 1
                    col[j] += 1

        for i in range(n):
            for j in range(m):
                if grid[i][j] and row[i] > 0 and col[j] > 0:
                    res += row[i]*col[j]

        return res
