# 有点像直接统计答案，直接枚举所有字母

```python
class Solution:
    def countPalindromicSubsequence(self, s: str) -> int:
        res = 0
        for c in ascii_lowercase:
            i = s.find(c)
            if i < 0:
                continue
            j = s.rfind(c)
            res += len(set(s[i+1:j]))
        return res

```

# 枚举中间

```python
class Solution:
    def countPalindromicSubsequence(self, s: str) -> int:
        res = set()
        pre = set()
        suf = Counter(s[1:])

        for i in range(1,len(s) - 1):
            mid = s[i]
            suf[mid] -= 1
            if suf[mid] == 0:
                del suf[mid ]
            pre.add(s[i-1])
            for c in pre & suf.keys():
                res.add(c + mid)
        return len(res)
```
