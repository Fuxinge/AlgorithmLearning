# 枚举左，维护右

```python
class Solution:
    def maximumTripletValue(self, nums: List[int]) -> int:
        res = 0
        prema = 0
        diff = 0
        for k,c in enumerate(nums):
            res = max(res,c * diff)
            diff = max(diff,prema - c)
            prema = max(prema,c)

        return res

```

# 枚举中间

```python
class Solution:
    def maximumTripletValue(self, nums: List[int]) -> int:
        res = 0
        n = len(nums)
        suf = [nums[-1]] * n
        pre = [nums[0]] * n
        for i in range(n - 2, -1, -1):
            suf[i] = max(suf[i+1],nums[i])
        for i in range(1,n):
            pre[i] = max(pre[i-1],nums[i])

        for i in range(1,n-1):
            if pre[i-1] > nums[i]:
                res = max(res,(pre[i-1] - nums[i]) * suf[i + 1])

        return res
```
