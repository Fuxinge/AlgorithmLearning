class Solution:
    def numberOfBoomerangs(self, points: List[List[int]]) -> int:
        n = len(points)
        if n < 3:
            return 0
        res = 0
        def cal(a,b,c,d):
            return (a-c)**2 + (b-d) **2

        for a,b in points:
            cnt = defaultdict(int)
            for c,d in points:
                d2 = cal(a,b,c,d)
                res += cnt[d2] * 2
                cnt[d2] += 1

        return res
