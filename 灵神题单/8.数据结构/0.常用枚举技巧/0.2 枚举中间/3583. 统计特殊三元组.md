```python
class Solution:
    def specialTriplets(self, nums: List[int]) -> int:
        mod = int(1e9 + 7)
        suf = Counter(nums)
        res = 0
        pre = defaultdict(int )
        for i,c in enumerate(nums):
            suf[c] -= 1
            if suf[c * 2] != 0 and c*2 in pre :
                res += suf[c * 2] * pre[c * 2]
                res %= mod
            pre[c] += 1

        return res
```
