class Solution:
    def findDiagonalOrder(self, grid: List[List[int]]) -> List[int]:
        m,n = len(grid),len(grid[0])
        grid = [row[::-1] for row in grid]
        res = []
        for k in range(1,n + m):
            min_j = max(0,n - k)
            max_j = min(n - 1,m + n - 1 - k)
            s = [grid[k + j - n][j] for j in range(min_j,max_j + 1)]
            if k % 2 == 1:
                s.reverse()
            res += s
        return res
