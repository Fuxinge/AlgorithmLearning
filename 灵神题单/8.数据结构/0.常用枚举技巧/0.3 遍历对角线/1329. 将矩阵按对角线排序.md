```python
class Solution:
    def diagonalSort(self, grid: List[List[int]]) -> List[List[int]]:
        m,n = len(grid),len(grid[0])
        # k = i - j  + n
        for k in range(1,m+n):
            min_j = max(n - k, 0)
            max_j = min(n - 1, m - 1 + n - k)
            s = [grid[k + j - n][j] for j in range(min_j,max_j + 1)]
            s.sort()
            for j,val in zip(range(min_j,max_j + 1),s):
                grid[k - n + j][j] = val

        return grid
```
