class Solution:
    def differenceOfDistinctValues(self, grid: List[List[int]]) -> List[List[int]]:
        m,n = len(grid),len(grid[0])
        res = [[0] * n for _ in range(m)]
        for k in range(1,m+n):
            # k = i - j + n 这里的n必须是列数,上道题是因为m=n
            min_j = max(0,n - k)
            max_j = min(n - 1, m - 1 + n - k)
            for j in range(min_j,max_j + 1):
                mpup = len(set([grid[k + j0 - n][j0] for j0 in range(min_j,j)]))
                mpdown = len(set([grid[k + j0 - n][j0] for j0 in range(j+1,max_j+1)]))
                res[k + j - n][j] = abs(mpdown - mpup)

        return res
